238 private static APClass getUnique(String rule, int subClass, BondType bt)
239 {
240 APClass newApc = new APClass();
241 synchronized (uniqueAPClassesLock)
242 {
243 boolean found = false;
244 for (APClass existingApc : uniqueAPClasses)
245 {
246 if (existingApc.getRule().equals(rule)
247 && existingApc.getSubClass()==subClass)
248 {
249 newApc = existingApc;
250 found = true;
251 break;
252 }
253 }
254 if (!found)
255 {
256 newApc.setRule(rule);
257 newApc.setSubClass(subClass);
258 newApc.setBondType(bt);
259 } else {
260 // NB: the default bond type for RCAs must not be changed, but
261 // For non-RCA APClasses we do update the bond type.
262 if (bt != newApc.bndTyp && !RCAAPCLASSSET.contains(newApc))
263 {
264 System.err.println("WARNING! Changing bond order of "
265 + "APClass " + newApc + ": " + newApc.bndTyp
266 + " -> " + bt);
267 newApc.setBondType(bt);
268 }
269 }
270 uniqueAPClasses.add(newApc);
271 }
272 return newApc;
273 }

408 public static boolean isValidAPClassString(String s)
409 {
410 if (!s.matches("^[a-z,A-Z,0-9].*"))
411 return false;
412
413 if (!s.matches(".*[0-9]$"))
414 return false;
415
416 if (s.contains(" "))
417 return false;
418
419 if (!s.contains(DENOPTIMConstants.SEPARATORAPPROPSCL))
420 return false;
421
422 int numSep = 0;
423 for (int i=0; i<s.length(); i++)
424 {
425 if (s.charAt(i) == DENOPTIMConstants.SEPARATORAPPROPSCL.charAt(0))
426 {
427 numSep++;
428 }
429 }
430 if (numSep != 1)
431 return false;
432
433 String apRule = s.split(DENOPTIMConstants.SEPARATORAPPROPSCL)[0];
434 String sub = s.split(DENOPTIMConstants.SEPARATORAPPROPSCL)[1];
435
436 return APClass.isValidAPRuleString(apRule)
437 && APClass.isValidAPSubCLassString(sub);
438 }

496 @Override
497 public APClass deserialize(JsonElement json, Type typeOfT,
498 JsonDeserializationContext context) throws JsonParseException
499 {
500 JsonObject jo = json.getAsJsonObject();
501 APClass apc = null;
502 if (jo.has("bndTyp"))
503 {
504 apc = getUnique(jo.get("rule").getAsString(),
505 jo.get("subClass").getAsInt(),
506 context.deserialize(jo.get("bndTyp"),BondType.class));
507 } else {
508 //Only for conversion to V3
509 String rule = jo.get("rule").getAsString();
510 int subClass = jo.get("subClass").getAsInt();
511 boolean found = false;
512 for (APClass existingApc : uniqueAPClasses)
513 {
514 if (existingApc.getRule().equals(rule)
515 && existingApc.getSubClass()==subClass)
516 {
517 apc = existingApc;
518 found = true;
519 break;
520 }
521 }
522 if (!found)
523 {
524 System.out.println("WARNING! Setting " + DEFAULTBT
525 + " for "+rule+":"+subClass);
526 apc = getUnique(jo.get("rule").getAsString(),
527 jo.get("subClass").getAsInt(), DEFAULTBT);
528 }
529 }
530 return apc;
