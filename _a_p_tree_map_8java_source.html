101 .filter(k -> k.compareTo(key)==0)
102 .findFirst()
103 .orElse(null);
104 if (possiblyOverlappingKey != null
105 && possiblyOverlappingKey != key)
106 {
107 try
108 {
109 GraphUtils.resetUniqueAPCounter(maxID+1);
110 } catch (DENOPTIMException e)
111 {
112 // This happens if the atomic integer behind that is
113 // behind GraphUtils.getUniqueAPIndex() is already ahead of
114 // maxID, in which case we do not need to reset it.
115 }
116 key.setID(GraphUtils.getUniqueAPIndex());
117 }
118 if (key.getID()>maxID)
119 maxID = key.getID();
120 return super.put(key, value);
