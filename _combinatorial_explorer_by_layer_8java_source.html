121 {
122 this.settings = settings;
123 this.verbosity = settings.getVerbosity();
124 this.restartFromChkPt = settings.restartFromCheckPoint();

126 fsSettings = new FragmentSpaceParameters();
127 if (settings.containsParameters(ParametersType.FS_PARAMS))
128 {
129 fsSettings = (FragmentSpaceParameters)settings.getParameters(
130 ParametersType.FS_PARAMS);

133 futures = new ArrayList<>();
134 submitted = new ArrayList<>();
135
136 tpe = new ThreadPoolExecutor(settings.getNumberOfCPU(),
137 settings.getNumberOfCPU(),
138 Long.MAX_VALUE,
139 TimeUnit.NANOSECONDS,
140 new ArrayBlockingQueue<Runnable>(1));
141
142 Runtime.getRuntime().addShutdownHook(new Thread()
143 {
144 @Override
145 public void run()
146 {
147 tpe.shutdown(); // Disable new tasks from being submitted
148 try
149 {
150 // Wait a while for existing tasks to terminate
151 if (!tpe.awaitTermination(30, TimeUnit.SECONDS))
152 {
153 tpe.shutdownNow(); // Cancel currently executing tasks
154 }
155
156 if (!tpe.awaitTermination(60, TimeUnit.SECONDS))
157 {
158 // pool didn't terminate after the second try
159 }
160 }
161 catch (InterruptedException ie)
162 {
163 cleanup(tpe, futures, submitted);
164 // (Re-)Cancel if current thread also interrupted
165 tpe.shutdownNow();
166 // Preserve interrupt status
167 Thread.currentThread().interrupt();
168 }
169 }
170 });
171
172 // by default the ThreadPoolExecutor will throw an exception
173 tpe.setRejectedExecutionHandler(new RejectedExecutionHandler()
174 {
175 @Override
176 public void rejectedExecution(Runnable r,
177 ThreadPoolExecutor executor)
178 {
179 try
180 {
181 // this will block if the queue is full
182 executor.getQueue().put(r);
183 }
184 catch (InterruptedException ex)
185 {
186 if (verbosity > 1)
187 {
188 ex.printStackTrace();
189 String msg = "EXCEPTION in rejectedExecution.";
190 settings.getLogger().log(Level.WARNING,msg);
191 }
192 }
193 }
194 });
195 }

262 private void makeCheckPoint() throws DENOPTIMException
263 {
264 for (int i=0; i<submitted.size(); i++)
265 {
266 GraphBuildingTask tsk = submitted.get(i);
267 int idToChkPt = -1;
268 boolean storeChkPt = false;
269 if (!tsk.isCompleted() && i>0)
270 {
271 if (submitted.get(i-1).isCompleted())
272 {
273 // The last completed task preceeding the first uncompleted
274 idToChkPt = i-1;
275 storeChkPt = true;
276 }
277 }
278 else if (tsk.isCompleted() && (i==submitted.size()-1))
279 {
280 idToChkPt = i;
281 storeChkPt = true;
282 }
283 if (storeChkPt)
284 {
285 tsk = submitted.get(idToChkPt);
286 CheckPoint chk = settings.getCheckPoint();
287 chk.setSafelyCompletedGraphId(tsk.getGraphId());
288 chk.setRootId(tsk.getRootId());
289 chk.setNextIds(tsk.getNextIds());
290 chk.setLevel(tsk.getLevel());
291 chk.setUnqVrtId(GraphUtils.getUniqueVertexIndex());
292 chk.setUnqGraphId(GraphUtils.getUniqueGraphIndex());
293 chk.setUnqMolId(GraphUtils.getUniqueMoleculeIndex());
294 CEBLUtils.serializeCheckPoint(settings);
295 break;
296 }
297 }
298 }

325 public void run() throws DENOPTIMException
326 {
327 String msg = "";
328 StopWatch watch = new StopWatch();
329 watch.start();
330
331 tpe.prestartAllCoreThreads();
332
333 int level = -1;
334 if (restartFromChkPt)
335 {
336 firstAfterRestart = true;
337 CheckPoint chk = settings.getCheckPoint();
338 level = chk.getLevel();
339 GraphUtils.resetUniqueVertexCounter(chk.getUnqVrtId());
340 GraphUtils.resetUniqueGraphCounter(chk.getUnqGraphId());
341 GraphUtils.resetUniqueMoleculeCounter(chk.getUnqMolId());
342
343 msg = "Restarting Combinatorial job from checkpoint file. "
344 + DENOPTIMConstants.EOL
345 + "All graphs with ID higher than "
346 + chk.getLatestSafelyCompletedGraphId()
347 + " are now being re-generated. Be aware that this "
348 + "includes graphs managed either by "
349 + "partially executed or completed tasks of the previous "
350 + "run. "
351 + DENOPTIMConstants.EOL
352 + "To avoid duplicates, you should remove from the results "
353 + "of the previos run all graphs with ID higher than "
354 + chk.getLatestSafelyCompletedGraphId()
355 + ". You can find them in the "
356 + "index file ('" + DENOPTIMConstants.FSEIDXNAMEROOT + level
357 + ".txt')."
358 + DENOPTIMConstants.EOL
359 + "Now reading '" + DENOPTIMConstants.SERGFILENAMEEXT + "' "
360 + "files from '"
361 + CEBLUtils.getNameOfStorageDir(settings, level) + "'.";
362 settings.getLogger().log(Level.WARNING,msg);
363
364 if (!denoptim.files.FileUtils.checkExists(
365 CEBLUtils.getNameOfStorageDir(settings, level)))
366 {
367 msg = "ERROR! Folder '"
368 + CEBLUtils.getNameOfStorageDir(settings, level)
369 + "' does not exist! Use 'FSE-DBROOTFOLDER' to "
370 + "provide the pathname to the existing folder where "
371 + "the previously generated graphs are located.";
372 settings.getLogger().log(Level.SEVERE,msg);
373 throw new DENOPTIMException(msg);
374 }
375
376 Collection<File> lst = FileUtils.listFiles(
377 new File(CEBLUtils.getNameOfStorageDir(settings, level)),
378 new String[] {DENOPTIMConstants.SERGFILENAMEEXT},false);
379 // Keep only safely completed serialized graphs
380 serFromChkRestart = lst.size();
381 for (File f : lst)
382 {
383 String fName = f.getName();
384 int serGrphID = Integer.parseInt(fName.substring(
385 DENOPTIMConstants.SERGFILENAMEROOT.length(),
386 fName.length()
387 - DENOPTIMConstants.SERGFILENAMEEXT.length()-1));
388 if (serGrphID > chk.getLatestSafelyCompletedGraphId())
389 {
390 msg = "Removing non-safely completed graph '" + fName + "'";
391 settings.getLogger().log(Level.WARNING,msg);
392 serFromChkRestart--;
393 denoptim.files.FileUtils.deleteFile(
394 CEBLUtils.getNameOfStorageDir(settings, level)
395 + File.separator + fName);
396 }
397 }
398 }
399
400 boolean interrupted = false;
401 while (level <= settings.getMaxLevel())
402 {
403 msg = "Starting exploration of level " + level;
404 settings.getLogger().log(Level.INFO,msg);
405
406 int numSubTasks = exploreCombinationsAtGivenLevel(level);
407
408 if (numSubTasks > 0)
409 {
410 // wait for pending tasks to finish
411 long startTime = System.currentTimeMillis();
412 while (true)
413 {
414 long endTime = System.currentTimeMillis();
415 long millis = (endTime - startTime);
416 if (allTasksCompleted())
417 {
418 Collection<File> lst = FileUtils.listFiles(
419 new File(CEBLUtils.getNameOfStorageDir(
420 settings, level)),
421 new String[]{DENOPTIMConstants.SERGFILENAMEEXT},
422 false);
423 int outCount = lst.size() - serFromChkRestart;
424 int totSubmSubTasks = countSubTasks();
425 if (outCount != totSubmSubTasks && level > -1)
426 {
427 msg = "Mismatch between the number of submitted "
428 + "tasks (" + totSubmSubTasks + ") and those "
429 + "listed in "
430 + CEBLUtils.getNameOfStorageDir(settings, level)
431 + "(" + outCount + ")";
432 settings.getLogger().log(Level.SEVERE,msg);
433 throw new DENOPTIMException(msg);
434 }
435 break;
436 }
437 else
438 {
439 if (subtaskHasException())
440 {
441 stopRun();
442 msg = "Exception in submitted task.";
443 throw new DENOPTIMException(msg,thrownByTask);
444 }
445 }
446
447 if (verbosity > 0)
448 {
449 msg = "Waiting for completion of level " + level;
450 msg = msg + String.format(" (elapsed %d min, %d sec)",
451 TimeUnit.MILLISECONDS.toMinutes(millis),
452 TimeUnit.MILLISECONDS.toSeconds(millis)
453 - TimeUnit.MINUTES.toSeconds(
454 TimeUnit.MILLISECONDS.toMinutes(millis)));
455 settings.getLogger().log(Level.INFO,msg);
456 }
457
458 if (millis > settings.getMaxWait())
459 {
460 stopRun();
461 msg = "Timeout reached: stopping all subtasks.";
462 settings.getLogger().log(Level.SEVERE, msg);
463 interrupted = true;
464 break;
465 }
466
467 try
468 {
469 Thread.sleep(settings.getWaitStep());
470 }
471 catch (Throwable t)
472 {
473 throw new DENOPTIMException(t);
474 }
475 }
476 }
477
478 // Clean queue
479 cleanup(tpe, futures, submitted);
480
481 // Level and all tasks completed
482 if (interrupted)
483 {
484 break;
485 }
486 else
487 {
488 msg = "Exploration of level " + level + " "
489 + "completed" + DENOPTIMConstants.EOL
490 + "----------------------------------------"
491 + "----------------------------------------"
492 + DENOPTIMConstants.EOL;
493 settings.getLogger().log(Level.INFO,msg);
494 }
495
496 // Increment level index
497 level++;
498
499 // Check for possibility of building another level
500 boolean noRoot = false;
501 if (numSubTasks == 0)
502 {
503 // Needed to perceive prev.lev. when restarting from checkpoint
504 try
505 {
506 Collection<File> lstRootsForNextLev = FileUtils.listFiles(
507 new File(CEBLUtils.getNameOfStorageDir(settings,
508 level-1)),
509 new String[] {DENOPTIMConstants.SERGFILENAMEEXT},
510 false);
511 if (lstRootsForNextLev.size() == 0)
512 {
513 noRoot = true;
514 }
515 }
516 catch (Throwable t)
517 {
518 noRoot = true;
519 }
520 }
521 if (noRoot)
522 {
523 msg = "Previous level did not return any extendable graph. "
524 + "Terminating exploration." + DENOPTIMConstants.EOL
525 + "----------------------------------------"
526 + "----------------------------------------"
527 + DENOPTIMConstants.EOL;
528 settings.getLogger().log(Level.INFO,msg);
529 break;
530 }
531
532 if (restartFromChkPt)
533 {
534 restartFromChkPt = false;
535 }
536 }
537
538 // shutdown thread pool
539 tpe.shutdown();
540
541 // closing messages
542 watch.stop();
543 msg = "Overall time: " + watch.toString() + ". "
544 + DENOPTIMConstants.EOL
545 + "FragSpaceExplorer run completed." + DENOPTIMConstants.EOL;
546 settings.getLogger().log(Level.INFO, msg);
547 }

560 private int exploreCombinationsAtGivenLevel(int level)
561 throws DENOPTIMException
562 {
563 // The very first level only has the scaffold or the user defined roots
564 if (level == -1)
565 {
566 ArrayList<DGraph> scafLevel = new ArrayList<DGraph>();
567 if (settings.useGivenRoots())
568 {
569 // User defined root graphs
570 if (verbosity > 0)
571 {
572 String msg = "User defined root graphs are treated as "
573 + "if they were scaffolds. "
574 + "That is, the 'level' of the available attachment "
575 + "points is considered to be '-1' "
576 + "no matter what is the actual level of such APs "
577 + "in the root graph.";
578 settings.getLogger().log(Level.WARNING, msg);
579 }
580 for (DGraph rootGraph : settings.getRootGraphs())
581 {
582 // Vertex ID in root can be whatever and we ignore them
583 // when setting new vertex IDs. To do this, we change sign
584 // to the old IDs to avoid them clashing with the new ones,
585 rootGraph.changeSignOfVertexID();
586 // ...and re-assign all IDs
587 rootGraph.renumberGraphVertices();
588 rootGraph.setGraphId(GraphUtils.getUniqueGraphIndex());
589 scafLevel.add(rootGraph);
590 }
591 }
592 else
593 {
594 // Make the root graphs from the scaffolds
595 for (int i=0; i<fsSettings.getFragmentSpace()
596 .getScaffoldLibrary().size(); i++)
597 {
598 scafLevel.add(startNewGraphFromScaffold(i));
599 }
600 }
601
602 // Store them
603 CEBLUtils.storeAllGraphsOfLevel(settings, scafLevel,level);
604
605 // All done
606 return scafLevel.size();
607 }
608
609 // Iterate through the previous level
610 String msg = "";
611 int numSubTasks = 0;
612 int cntRoot = 0;
613 int total = 0;
614 int itersFromChkPt = 0;
615 String prevLevDirName = CEBLUtils.getNameOfStorageDir(settings,level-1);
616 if (!denoptim.files.FileUtils.checkExists(prevLevDirName))
617 {
618 msg = "Previous level folder '" + prevLevDirName + "' not found!";
619 throw new DENOPTIMException(msg);
620 }
621 Collection<File> files = FileUtils.listFiles(new File(prevLevDirName),
622 new String[] {DENOPTIMConstants.SERGFILENAMEEXT}, false);
623 ArrayList<File> lstFiles = new ArrayList<File>(files);
624 Collections.sort(lstFiles);
625 for (File file : lstFiles)
626 {
627 cntRoot++;
628 if (restartFromChkPt &&
629 settings.getCheckPoint().serFileAlreadyUsed(file.getName()))
630 {
631 continue;
632 }
633
634 DGraph rootGraph = DenoptimIO.readDENOPTIMGraphsFromJSONFile(
635 file.getAbsolutePath()).get(0);
636
637 // Get combination factory
638 FragsCombinationIterator fcf = new FragsCombinationIterator(
639 fsSettings, rootGraph);
640
641 if (restartFromChkPt && firstAfterRestart)
642 {
643 firstAfterRestart = false;
644 fcf.setStartingPoint(settings.getCheckPoint()
645 .getNextIds());
646 }
647
648 // Print summary
649 if (settings.getLogger().isLoggable(Level.FINE))
650 {
651 StringBuilder sb = new StringBuilder(512);
652 sb.append("Root: " + file.getName() + DENOPTIMConstants.EOL);
653 sb.append(" - #Usable APs on root = ");
654 sb.append(fcf.getNumRootAPs() + DENOPTIMConstants.EOL);
655 sb.append(" - Size of candidates sets = ");
656 sb.append(fcf.getSizesOfCandidateSets()+DENOPTIMConstants.EOL);
657 sb.append(" - Total #Combinations = ");
658 sb.append(fcf.getTotNumbCombs() + DENOPTIMConstants.EOL);
659 sb.append(" - Root graph: " + DENOPTIMConstants.EOL+rootGraph);
660 sb.append(DENOPTIMConstants.EOL);
661 sb.append(" - Details for root APs:");
662 sb.append(DENOPTIMConstants.EOL);
663 Map<IdFragmentAndAP, ArrayList<IdFragmentAndAP>> m =
664 fcf.getCandidatesMap();
665 for (IdFragmentAndAP srcAP : m.keySet())
666 {
667 sb.append(" -> "+srcAP).append(DENOPTIMConstants.EOL);
668 int iTrgAP = 0;
669 for (IdFragmentAndAP trgAP : m.get(srcAP))
670 {
671 iTrgAP++;
672 sb.append(" " + iTrgAP + " -> "+trgAP);
673 sb.append(DENOPTIMConstants.EOL);
674 }
675 }
676 settings.getLogger().log(Level.FINE, sb.toString()
677 + DENOPTIMConstants.EOL);
678 }
679
680 // Iterate over all combinations
681 try
682 {
683 while (fcf.hasNext())
684 {
685 if (subtaskHasException())
686 {
687 stopRun();
688 msg = "Exception in submitted task.";
689 throw new DENOPTIMException(msg,thrownByTask);
690 }
691
692 FragsCombination fragsToAdd = fcf.next();
693
694 GraphBuildingTask task = new GraphBuildingTask(
695 settings,
696 rootGraph, fragsToAdd, level,
697 settings.getWorkDirectory(),
698 settings.getVerbosity());
699
700 ArrayList<Integer> nextIds = fcf.getNextIds();
701 task.setNextIds(nextIds);
702
703 submitted.add(task);
704 futures.add(tpe.submit(task));
705 numSubTasks++;
706 if (itersFromChkPt >= settings.getCheckPointStep())
707 {
708 itersFromChkPt = 0;
709 makeCheckPoint();
710 }
711 itersFromChkPt++;
712
713 // Code meant only for preparation of checkpoint files
714 // The two following variables define at which point in the
715 // exploration of the space we want to stop.
716 int maxL = 2;
717 int maxI = 50;
718 if (settings.prepareFilesForTests())
719 {
720 System.out.println("Wait until "+level+"=="+maxL+" and "
721 +(total+fcf.getNumGeneratedCombs())+"=="+maxI);
722 if (level>=maxL &&
723 (total+fcf.getNumGeneratedCombs()>=maxI))
724 {
725 System.out.println("Execution stopped: now waiting "
726 + " for checkpoint file to mature");
727 int iWait=0;
728 int nEqual = 0;
729 ArrayList<Integer> oldIds =new ArrayList<Integer>();
730 ArrayList<Integer> nowIds =new ArrayList<Integer>();
731 makeCheckPoint();
732 oldIds.addAll(settings.getCheckPoint().getNextIds());
733 while (true)
734 {
735 iWait++;
736 Thread.sleep(1000); // in millisec
737 if (iWait > 120)
738 {
739 System.out.println("NOT CONVERGED");
740 throw new DENOPTIMException("Generation of "
741 + "checkpoint file did not "
742 + "converge.");
743 }
744 makeCheckPoint();
745 nowIds.clear();
746 nowIds.addAll(
747 settings.getCheckPoint().getNextIds());
748 System.out.println(oldIds + " " + nowIds
749 + " nEqualChecks:" + nEqual);
750 boolean converged = true;
751 if (nowIds.size() != oldIds.size())
752 {
753 converged = false;
754 }
755 else
756 {
757 for (int iId=0; iId<nowIds.size(); iId++)
758 {
759 if (nowIds.get(iId) != oldIds.get(iId))
760 {
761 nEqual = 0;
762 converged = false;
763 break;
764 }
765 }
766 }
767 if (!converged)
768 {
769 oldIds.clear();
770 oldIds.addAll(nowIds);
771 continue;
772 }
773 nEqual++;
774 if (nEqual >= 10)
775 {
776 break;
777 }
778 }
779 System.out.println("Stopped with converged "
780 + "checkpoint IDs: " + nowIds);
781 throw new DENOPTIMException("Stopping due to "
782 + "request of generating checkpoint data "
783 + "for testing purposes");
784 }
785 }
786 }
787 }
788 catch (DENOPTIMException dex)
789 {
790 cleanup(tpe, futures, submitted);
791 tpe.shutdown();
792 throw dex;
793 }
794 catch (Exception ex)
795 {
796 cleanup(tpe, futures, submitted);
797 tpe.shutdown();
798 throw new DENOPTIMException(ex);
799 }
800
801 msg = fcf.getNumGeneratedCombs() + "/"
802 + fcf.getTotNumbCombs() + " combination generated "
803 + "for level " + level + " of graph " + cntRoot;
804 settings.getLogger().log(Level.FINE, msg);
805 total = total + fcf.getNumGeneratedCombs();
806 }
807
808 msg = "Total number of combination of fragments generated "
809 + "for level " + level + " = " + total;
810 settings.getLogger().log(Level.INFO, msg);
811
812/*
813 // Code meant only to generate checkpoint file at the end of a level
814 if (settings.prepareFilesForTests())
815 {
816 System.out.println("Last submissions for level "+level+". Wait for "
817 + "completion of the tasks.");
818 while (true)
819 {
820 makeCheckPoint();
821 System.out.println("Ctrl-Z once converged: " +
822 settings.getCheckPoint().getNextIds());
823 GenUtils.pause();
824 }
825 }
826*/
827
828 return numSubTasks;
829 }
