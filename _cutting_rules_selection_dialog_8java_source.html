369 @Override
370 public void actionPerformed(ActionEvent e)
371 {
372 DefaultTableModel chosenTab = defaultRulesTabModel;
373 if (rdbUseCustom.isSelected())
374 chosenTab = customRulesTabModel;
375
376 boolean setOfCuttingRulesHasBeenModified = false;
377
378 List<CuttingRule> chosenOnes = new ArrayList<CuttingRule>();
379 for (int iRow=0; iRow<chosenTab.getRowCount(); iRow++)
380 {
381 String[] values = new String[5];
382 for (int iCol=0; iCol<5; iCol++)
383 {
384 values[iCol] = chosenTab.getValueAt(iRow,iCol)
385 .toString().trim();
386 }
387 ArrayList<String> opts = new ArrayList<String>();
388 if (!chosenTab.getValueAt(iRow,5).toString().trim().isBlank())
389 opts.addAll(Arrays.asList(chosenTab.getValueAt(iRow,5)
390 .toString().trim().split("\\s+")));
391
392 int prioIdx = 0;
393 try {
394 prioIdx = Integer.parseInt(values[1]);
395 } catch (Throwable t)
396 {
397 JOptionPane.showMessageDialog(btnDone,String.format(
398 "<html><body width='%1s'>"
399 + "Could not convert string '" + values[1]
400 + "' into an integer. Ignoring row " + iRow
401 + " in table of custom cutting rules.</html>",
402 400),
403 "Error",
404 JOptionPane.ERROR_MESSAGE,
405 UIManager.getIcon("OptionPane.errorIcon"));
406 continue;
407 }
408
409 CuttingRule newCuttingRule = new CuttingRule(values[0],
410 values[2], values[4], values[3], prioIdx, opts);
411 chosenOnes.add(newCuttingRule);
412
413 // Verify introduction of differences
414 if (!setOfCuttingRulesHasBeenModified)
415 {
416 boolean fromDefault = false;
417 if (defaultCuttingRules.size() > iRow)
418 {
419 fromDefault = defaultCuttingRules.get(iRow).equals(
420 newCuttingRule);
421 }
422 boolean fromCustom = false;
423 if (customCuttingRules.size() > iRow)
424 {
425 fromCustom = customCuttingRules.get(iRow).equals(
426 newCuttingRule);
427 }
428 if (!fromDefault && !fromCustom)
429 {
430 setOfCuttingRulesHasBeenModified = true;
431 }
432 }
433 }
434
435 // Verify shortening of the list of rules (case not covered above)
436 if (chosenOnes.size() != defaultCuttingRules.size()
437 && chosenOnes.size() != customCuttingRules.size())
438 {
439 setOfCuttingRulesHasBeenModified = true;
440 }
441
442 //Store a copy of the modified/customized rules
443 useDefaultNextTime = false;
444 if (setOfCuttingRulesHasBeenModified)
445 {
446 try
447 {
448 lastUsedCutRulFile = nextWrittenCutRulFile;
449 DenoptimIO.writeCuttingRules(lastUsedCutRulFile,
450 chosenOnes);
451 FileUtils.addToRecentFiles(lastUsedCutRulFile,
452 FileFormat.CUTRULE);
453 } catch (DENOPTIMException e1)
454 {
455 e1.printStackTrace();
456 }
457 } else {
458 if (rdbUseDefault.isSelected())
459 useDefaultNextTime = true;
460 }
461
462 result = chosenOnes;
463 close();
464 }
465 });
466 getRootPane().setDefaultButton(btnDone);

604 private class MyJSVGCanvas extends JSVGCanvas
605 {
609 private static final long serialVersionUID = 1L;
610
611 public MyJSVGCanvas()
612 {
613 super();
614
615 // To change the interaction modes. First, disable the defaults
616 setEnableZoomInteractor(false);
617 setEnableImageZoomInteractor(false);
618 setEnablePanInteractor(false);
619 setEnableRotateInteractor(false);
620 setEnableResetTransformInteractor(false);
621
622 // Override the disabled default interactors that are still saved
623 // as fields.
624
625 // Then, we add our own interpreters of the mouse+key input
626
630 zoomInteractor = null;
631
635 imageZoomInteractor = new AbstractImageZoomInteractor() {
636 public boolean startInteraction(InputEvent ie) {
637 int mods = ie.getModifiersEx();
638 return
639 ie.getID() == MouseEvent.MOUSE_PRESSED &&
640 (mods & InputEvent.BUTTON1_DOWN_MASK) != 0 &&
641 (mods & InputEvent.SHIFT_DOWN_MASK) != 0;
642 }
643 };
644
648 panInteractor = new AbstractPanInteractor() {
649 public boolean startInteraction(InputEvent ie) {
650 int mods = ie.getModifiersEx();
651 return
652 ie.getID() == MouseEvent.MOUSE_PRESSED &&
653 (mods & InputEvent.BUTTON1_DOWN_MASK) != 0;
654 }
655
660 @Override
661 public void mouseExited(MouseEvent e) {}
662 };
663
667 rotateInteractor = null;
668
672 resetTransformInteractor = new AbstractResetTransformInteractor() {
673 public boolean startInteraction(InputEvent ie) {
674 int mods = ie.getModifiersEx();
675 return
676 ie.getID() == MouseEvent.MOUSE_PRESSED &&
677 (mods & InputEvent.CTRL_DOWN_MASK) != 0;
678 }
679 };
680 // Finally, enable the desired interactors. It is crucial to do this
681 // in the right order because the interactors are added to a List
682 // that retains the order and the events are interpreted so that
683 // the first interactor that returns true from startInteraction
684 // does the action, the others are not tested. Therefore, pan
685 // interactor, which has the less restrictive mask, must go last.
686 setEnableImageZoomInteractor(true);
687 setEnableResetTransformInteractor(true);
688 setEnablePanInteractor(true);
689 }
690 }

698 private String escapeCharactersForSMARTSViewer(String smarts)
699 {
700 String escapedSmarts = smarts;
701 escapedSmarts = escapedSmarts.replaceAll("%", "%25");
702 escapedSmarts = escapedSmarts.replaceAll("&", "%26");
703 escapedSmarts = escapedSmarts.replaceAll("\\+", "%2B");
704 escapedSmarts = escapedSmarts.replaceAll("#", "%23");
705 escapedSmarts = escapedSmarts.replaceAll(";", "%3B");
706
707 return escapedSmarts;
708 }
