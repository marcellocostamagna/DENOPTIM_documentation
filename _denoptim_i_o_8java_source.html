147 public static Object readDENOPTIMData(String pathname)
148 throws Exception
149 {
150 Object data = null;
151 File file = new File(pathname);
152 FileFormat format = FileFormat.UNRECOGNIZED;
153 format = FileUtils.detectFileFormat(file);
154 if (format == FileFormat.UNRECOGNIZED)
155 {
156 throw new UndetectedFileFormatException(file);
157 }
158 switch (format)
159 {
160 case VRTXSDF:
161 {
162 ArrayList<Vertex> vrtxs = readVertexes(file, BBType.FRAGMENT);
163 if (vrtxs.size()>1)
164 data = vrtxs;
165 else
166 data = vrtxs.get(0);
167 break;
168 }
169
170 case VRTXJSON:
171 {
172 ArrayList<Vertex> vrtxs = readVertexes(file, BBType.FRAGMENT);
173 if (vrtxs.size()>1)
174 data = vrtxs;
175 else
176 data = vrtxs.get(0);
177 break;
178 }
179
180 case CANDIDATESDF:
181 {
182 ArrayList<Candidate> cands = readCandidates(file, true);
183 if (cands.size()>1)
184 data = cands;
185 else
186 data = cands.get(0);
187 break;
188 }
189
190 case GRAPHSDF:
191 {
192 ArrayList<DGraph> graphs = readDENOPTIMGraphsFromFile(file,
193 format);
194 if (graphs.size()>1)
195 data = graphs;
196 else
197 data = graphs.get(0);
198 break;
199 }
200
201 case GRAPHJSON:
202 {
203 ArrayList<DGraph> graphs = readDENOPTIMGraphsFromFile(file,
204 format);
205 if (graphs.size()>1)
206 data = graphs;
207 else
208 data = graphs.get(0);
209 break;
210 }
211
212 case GENSUMMARY:
213 {
214 List<Candidate> cands = readGenerationFromSummary(file);
215 if (cands.size()>1)
216 data = cands;
217 else
218 data = cands.get(0);
219 break;
220 }
221
222 default:
223 throw new DENOPTIMException("Data from file of format '"
224 + format + "' cannot be loaded yet. Please, contact"
225 + "the development team.");
226 }
227 return data;
228 }

245 public static List<Candidate> readGenerationFromSummary(File file)
246 throws DENOPTIMException
247 {
248 ArrayList<Candidate> cands = new ArrayList<Candidate>();
249 List<String> pathnames = readPopulationMemberPathnames(file);
250 String genSummaryParentDir = file.getParent(); // GenXXX folder
251 for (String candPathname : pathnames)
252 {
253 File candFile = new File(candPathname);
254 if (candFile.exists())
255 {
256 cands.add(readCandidates(candFile).get(0));
257 } else if (genSummaryParentDir!=null) {
258 // Extract the possibly non-existing pathname where the
259 // candidates files were originally generated
260 String runFolderPathname =
261 candFile.getParentFile() // GenXXX
262 .getParent(); // RUN###
263 if (runFolderPathname==null)
264 throw new DENOPTIMException("Unable to find parent "
265 + "folder for '"+ genSummaryParentDir +"'");
266
267 String genAndMolPath = candFile.getAbsolutePath()
268 .substring(runFolderPathname.length());
269
270 cands.add(readCandidates(new File(
271 genSummaryParentDir+genAndMolPath)).get(0));
272 }
273 }
274 return cands;
275 }

290 public static List<IAtomContainer> readAllAtomContainers(File file)
291 throws IOException, CDKException
292 {
293 List<IAtomContainer> results = null;
294
295 FileReader formatReader = new FileReader(file);
296 IChemFormat chemFormat = new FormatFactory().guessFormat(
297 new BufferedReader(formatReader));
298 formatReader.close();
299
300 if (chemFormat instanceof CIFFormat)
301 {
303 // WARNING
305 //
306 // * CDK's CIFReader is broken (skips lines _audit, AND ignores
307 // connectivity)
308 // * BioJava's fails on CIF files that do not complain to mmCIF
309 // format. Try this:
310 // CifStructureConverter.fromPath(f.toPath());
311 //
312 // The workaround unit CDK's implementation is fixed, is to use Jmol
313 // to read the CIF and make an SDF that can then be read normally.
314 // In fact, Jmol's reader is more robust than CDK's and more
315 // versatile than BioJava's.
316 // Moreover it reads also the bonds defined in the CIF
317 // (which OpenBabel does not read).
318 // Yet, Jmol's methods are not visible and require spinning a
319 // a viewer.
320
321 Map<String, Object> info = new Hashtable<String, Object>();
322 info.put("adapter", new SmarterJmolAdapter());
323 info.put("isApp", false);
324 info.put("silent", "");
325 Viewer v = new Viewer(info);
326 v.loadModelFromFile(null, file.getAbsolutePath(), null, null,
327 false, null, null, null, 0, " ");
328 String tmp = FileUtils.getTempFolder() + FS + "convertedCIF.sdf";
329 v.scriptWait("write " + tmp + " as sdf");
330 v.dispose();
331
332 results = readAllAtomContainers(new File(tmp));
333 } else {
334 FileReader fileReader = new FileReader(file);
335 ReaderFactory readerFact = new ReaderFactory();
336 ISimpleChemObjectReader reader = readerFact.createReader(fileReader);
337 IChemFile chemFile = (IChemFile) reader.read(
338 (IChemObject) new ChemFile());
339 results = ChemFileManipulator.getAllAtomContainers(chemFile);
340 fileReader.close();
341 }
342
343 return results;
344 }

366 throw new DENOPTIMException(cdke);

368 try {

370 mdlreader.close();

373 throw new DENOPTIMException(ioe);
374 }
375 }
376
377 if (lstContainers.isEmpty()) {
378 throw new DENOPTIMException("No data found in " + fileName);
379 }
380
381 return lstContainers;

777 public static List<CandidateLW> readLightWeightCandidate(File file)
778 throws DENOPTIMException
779 {
780 List<String> propNames = new ArrayList<String>(Arrays.asList(
781 DENOPTIMConstants.UNIQUEIDTAG,
782 CDKConstants.TITLE
783 ));
784 List<String> optionalPropNames = new ArrayList<String>(Arrays.asList(
785 DENOPTIMConstants.FITNESSTAG,
786 DENOPTIMConstants.MOLERRORTAG,
787 DENOPTIMConstants.GMSGTAG,
788 DENOPTIMConstants.GRAPHLEVELTAG
789 ));
790 propNames.addAll(optionalPropNames);
791 List<Map<String, Object>> propsPerItem = readSDFProperties(
792 file.getAbsolutePath(), propNames);
793
794 List<CandidateLW> items = new ArrayList<CandidateLW>();
795 for (Map<String, Object> props : propsPerItem)
796 {
797 Object uidObj = props.get(DENOPTIMConstants.UNIQUEIDTAG);
798 if (uidObj==null )
799 {
800 throw new DENOPTIMException("Cannot create item if SDF tag "
801 + DENOPTIMConstants.UNIQUEIDTAG + " is null!");
802 }
803 Object nameObj = props.get(CDKConstants.TITLE);
804 if (nameObj==null)
805 {
806 throw new DENOPTIMException("Cannot create item is SDF tag "
807 + CDKConstants.TITLE + " is null!");
808 }
809 CandidateLW item = new CandidateLW(uidObj.toString(),
810 nameObj.toString(),file.getAbsolutePath());
811
812 for (String propName : optionalPropNames)
813 {
814 Object obj = props.get(propName);
815 if (obj != null)
816 {
817 switch (propName)
818 {
819 case DENOPTIMConstants.FITNESSTAG:
820 item.setFitness(Double.parseDouble(obj.toString()));
821 break;
822
823 case DENOPTIMConstants.MOLERRORTAG:
824 item.setError(obj.toString());
825 break;
826
827 case DENOPTIMConstants.GMSGTAG:
828 item.setGeneratingSource(obj.toString());
829 break;
830
831 case DENOPTIMConstants.GRAPHLEVELTAG:
832 item.setLevel(Integer.parseInt(obj.toString()));
833 break;
834 }
835 }
836 }
837 items.add(item);
838 }
839 return items;
840 }

852 public static double[] readPopulationProps(File file)
853 throws DENOPTIMException {
854 double[] vals = new double[4];
855 ArrayList<String> txt = readList(file.getAbsolutePath());
856 for (String line : txt) {
857 if (line.trim().length() < 8) {
858 continue;
859 }
860
861 String key = line.toUpperCase().trim().substring(0, 8);
862 switch (key) {
863 case ("MIN: "):
864 vals[0] = Double.parseDouble(line.split("\\s+")[1]);
865 break;
866
867 case ("MAX: "):
868 vals[1] = Double.parseDouble(line.split("\\s+")[1]);
869 break;
870
871 case ("MEAN: "):
872 vals[2] = Double.parseDouble(line.split("\\s+")[1]);
873 break;
874
875 case ("MEDIAN: "):
876 vals[3] = Double.parseDouble(line.split("\\s+")[1]);
877 break;
878 }
879 }
880 return vals;
881 }

1015 {
1016 List<Map<String,Object>> results = new ArrayList<Map<String,Object>>();
1017 ArrayList<IAtomContainer> iacs = DenoptimIO.readSDFFile(pathName);
