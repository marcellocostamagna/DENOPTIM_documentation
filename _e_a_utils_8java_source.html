187 public static int chooseNumberOfSitesToMutate(double[] multiSiteMutationProb,
188 double hit)
189 {
190 double tot = 0;
191 for (int i=0; i<multiSiteMutationProb.length; i++)
192 tot = tot + multiSiteMutationProb[i];
193
194 double scaledHit = hit * tot;
195
196 double min = 0;
197 double max = 0;
198 int choice = 0;
199 for (int i=0; i<multiSiteMutationProb.length; i++)
200 {
201 max = max + multiSiteMutationProb[i];
202 if (min < scaledHit && scaledHit <= max)
203 {
204 choice = i;
205 break;
206 }
207 min = Math.max(min,min+multiSiteMutationProb[i]);
208 }
209 return choice;
210 }

225 public static CandidateSource pickNewCandidateGenerationMode(
226 double xoverWeight, double mutWeight, double newWeight,
227 Randomizer randomizer)
228 {
229 double hit = randomizer.nextDouble()
230 * (xoverWeight + mutWeight + newWeight);
231 if (hit <= xoverWeight)
232 {
233 return CandidateSource.CROSSOVER;
234 } else if (xoverWeight < hit && hit <= (mutWeight+xoverWeight))
235 {
236 return CandidateSource.MUTATION;
237 } else {
238 return CandidateSource.CONSTRUCTION;
239 }

288 protected static Candidate buildCandidateByXOver(
289 ArrayList<Candidate> eligibleParents, Population population,
290 Monitor mnt, int[] choiceOfParents, int choiceOfXOverSites,
291 int choiceOfOffstring, GAParameters settings) throws DENOPTIMException
292 {
293 FragmentSpaceParameters fsParams = new FragmentSpaceParameters();
294 if (settings.containsParameters(ParametersType.FS_PARAMS))
295 {
296 fsParams = (FragmentSpaceParameters)settings.getParameters(
297 ParametersType.FS_PARAMS);
298 }
299 FragmentSpace fragSpace = fsParams.getFragmentSpace();
300
301 mnt.increase(CounterID.XOVERATTEMPTS);
302 mnt.increase(CounterID.NEWCANDIDATEATTEMPTS);
303
304 int numatt = 1;
305
306 // Identify a pair of parents that can do crossover, and a pair of
307 // vertexes from which we can define a subgraph (or a branch) to swap
308 XoverSite xos = null;
309 boolean foundPars = false;
310 while (numatt < settings.getMaxGeneticOpAttempts())
311 {
312 if (fragSpace.useAPclassBasedApproach())
313 {
314 xos = EAUtils.performFBCC(eligibleParents,
315 population, choiceOfParents, choiceOfXOverSites,
316 settings);
317 if (xos == null)
318 {
319 numatt++;
320 continue;
321 }
322 } else {
323 //TODO: make it reproducible using choiceOfParents and choiceOfXOverSites
324 Candidate[] parents = EAUtils.selectBasedOnFitness(
325 eligibleParents, 2, settings);
326 if (parents[0] == null || parents[1] == null)
327 {
328 numatt++;
329 continue;
330 }
331 //NB: this does not go into templates!
332 DGraph gpA = parents[0].getGraph();
333 List<Vertex> subGraphA = new ArrayList<Vertex>();
334 gpA.getChildrenTree(EAUtils.selectNonScaffoldNonCapVertex(
335 gpA, settings.getRandomizer()),subGraphA);
336
337 DGraph gpB = parents[1].getGraph();
338 List<Vertex> subGraphB = new ArrayList<Vertex>();
339 gpB.getChildrenTree(EAUtils.selectNonScaffoldNonCapVertex(
340 gpB, settings.getRandomizer()),subGraphB);
341 }
342 foundPars = true;
343 break;
344 }
345 mnt.increaseBy(CounterID.XOVERPARENTSEARCH, numatt);
346
347 if (!foundPars)
348 {
349 mnt.increase(CounterID.FAILEDXOVERATTEMPTS_FINDPARENTS);
350 mnt.increase(CounterID.FAILEDXOVERATTEMPTS);
351 return null;
352 }
353
354 Candidate cA = null, cB = null;
355 Vertex vA = null, vB = null;
356 vA = xos.getA().get(0);
357 vB = xos.getB().get(0);
358 DGraph gA = vA.getGraphOwner();
359 cA = gA.getOutermostGraphOwner().getCandidateOwner();
360 DGraph gB = vB.getGraphOwner();
361 cB = gB.getOutermostGraphOwner().getCandidateOwner();
362
363 String candIdA = cA.getName();
364 String candIdB = cB.getName();
365 int gid1 = gA.getGraphId();
366 int gid2 = gB.getGraphId();
367
368 // Start building the offspring
369 XoverSite xosOnClones = xos.projectToClonedGraphs();
370 DGraph gAClone = xosOnClones.getA().get(0).getGraphOwner();
371 DGraph gBClone = xosOnClones.getB().get(0).getGraphOwner();
372
373 try
374 {
375 if (!GraphOperations.performCrossover(xosOnClones,fragSpace))
376 {
377 mnt.increase(CounterID.FAILEDXOVERATTEMPTS_PERFORM);
378 mnt.increase(CounterID.FAILEDXOVERATTEMPTS);
379 return null;
380 }
381 } catch (Throwable t) {
382 if (!settings.xoverFailureTolerant)
383 {
384 t.printStackTrace();
385 ArrayList<DGraph> parents = new ArrayList<DGraph>();
386 parents.add(gA);
387 parents.add(gB);
388 DenoptimIO.writeGraphsToSDF(new File(settings.getDataDirectory()
389 + "_failed_xover.sdf"), parents, true,
390 settings.getLogger(), settings.getRandomizer());
391 throw new DENOPTIMException("Error while performing crossover! "+NL
392 + "XOverSite: " + xos.toString() + NL
393 + "XOverSite(C): " + xosOnClones.toString() + NL
394 + " Please, report this to the authors ",t);
395 }
396 mnt.increase(CounterID.FAILEDXOVERATTEMPTS_PERFORM);
397 mnt.increase(CounterID.FAILEDXOVERATTEMPTS);
398 return null;
399 }
400 gAClone.setGraphId(GraphUtils.getUniqueGraphIndex());
401 gBClone.setGraphId(GraphUtils.getUniqueGraphIndex());
402 String lstIdVA = "";
403 for (Vertex v : xos.getA())
404 lstIdVA = lstIdVA + "_" + v.getVertexId();
405 String lstIdVB = "";
406 for (Vertex v : xos.getB())
407 lstIdVB = lstIdVB + "_" + v.getVertexId();
408 String[] msgs = new String[2];
409 msgs[0] = "Xover: "
410 + "Gen:" + cA.getGeneration() + " Cand:" + candIdA
411 + "|" + gid1 + "|" + lstIdVA
412 + " X "
413 + "Gen:" + cB.getGeneration() + " Cand:" + candIdB
414 + "|" + gid2 + "|" + lstIdVB;
415 msgs[1] = "Xover: "
416 + "Gen:" + cB.getGeneration() + " Cand:" + candIdB
417 + "|" + gid2 + "|" + lstIdVB
418 + " X "
419 + "Gen:" + cA.getGeneration() + " Cand:" + candIdA
420 + "|" + gid1 + "|" + lstIdVA;
421
422 DGraph[] graphsAffectedByXover = new DGraph[2];
423 graphsAffectedByXover[0] = gAClone;
424 graphsAffectedByXover[1] = gBClone;
425
426 List<Candidate> validOffspring = new Population(settings);
427 for (int ig=0; ig<graphsAffectedByXover.length; ig++)
428 {
429 DGraph g = graphsAffectedByXover[ig];
430
431 // It makes sense to do this on the possibly embedded graph and not
432 // on their embedding owners because there cannot be any new cycle
433 // affecting the latter, but there can be ones affecting the first.
434 if (!EAUtils.setupRings(null, g, settings))
435 {
436 mnt.increase(CounterID.FAILEDXOVERATTEMPTS_SETUPRINGS);
437 continue;
438 }
439
440 // Finalize the graph that is at the outermost level
441 DGraph gOutermost = g.getOutermostGraphOwner();
442 gOutermost.addCappingGroups(fragSpace);
443 gOutermost.renumberGraphVertices();
444 gOutermost.setLocalMsg(msgs[ig]);
445
446 // Consider if the result can be used to define a new candidate
447 Object[] res = null;
448 try
449 {
450 res = gOutermost.checkConsistency(settings);
451 } catch (NullPointerException|IllegalArgumentException e)
452 {
453 if (!settings.xoverGraphFailedEvalTolerant)
454 {
455 ArrayList<DGraph> parents = new ArrayList<DGraph>();
456 parents.add(gA);
457 parents.add(gB);
458 parents.add(gAClone);
459 parents.add(gBClone);
460 DenoptimIO.writeGraphsToSDF(new File(settings.getDataDirectory()
461 + "_failed_xover-ed_check.sdf"), parents, true,
462 settings.getLogger(), settings.getRandomizer());
463 throw e;
464 } else {
465 res = null;
466 }
467 }
468 if (res != null)
469 {
470 if (!EAUtils.setupRings(res, gOutermost, settings))
471 {
472 mnt.increase(CounterID.FAILEDXOVERATTEMPTS_SETUPRINGS);
473 res = null;
474 }
475 } else {
476 mnt.increase(CounterID.FAILEDXOVERATTEMPTS_EVAL);
477 }
478
479 // Check if the chosen combination gives rise to forbidden ends
480 for (Vertex rcv : gOutermost.getFreeRCVertices())
481 {
482 APClass apc = rcv.getEdgeToParent().getSrcAP().getAPClass();
483 if (fragSpace.getCappingMap().get(apc)==null
484 && fragSpace.getForbiddenEndList().contains(apc))
485 {
486 mnt.increase(CounterID.FAILEDXOVERATTEMPTS_FORBENDS);
487 res = null;
488 }
489 }
490 if (res == null)
491 {
492 gOutermost.cleanup();
493 gOutermost = null;
494 continue;
495 }
496
497 // OK: we can now use it to make a new candidate
498 Candidate offspring = new Candidate(gOutermost);
499 offspring.setUID(res[0].toString().trim());
500 offspring.setSmiles(res[1].toString().trim());
501 offspring.setChemicalRepresentation((IAtomContainer) res[2]);
502
503 validOffspring.add(offspring);
504 }
505
506 if (validOffspring.size() == 0)
507 {
508 mnt.increase(CounterID.FAILEDXOVERATTEMPTS);
509 return null;
510 }
511
512 Candidate chosenOffspring = null;
513 if (choiceOfOffstring<0)
514 {
515 chosenOffspring = settings.getRandomizer().randomlyChooseOne(
516 validOffspring);
517 chosenOffspring.setName("M" + GeneralUtils.getPaddedString(
518 DENOPTIMConstants.MOLDIGITS,
519 GraphUtils.getUniqueMoleculeIndex()));
520 } else {
521 chosenOffspring = validOffspring.get(choiceOfOffstring);
522 }
523 return chosenOffspring;
524 }

528 protected static Candidate buildCandidateByMutation(
529 ArrayList<Candidate> eligibleParents, Monitor mnt,
530 GAParameters settings) throws DENOPTIMException
531 {
532 FragmentSpaceParameters fsParams = new FragmentSpaceParameters();
533 if (settings.containsParameters(ParametersType.FS_PARAMS))
534 {
535 fsParams = (FragmentSpaceParameters)settings.getParameters(
536 ParametersType.FS_PARAMS);
537 }
538 FragmentSpace fragSpace = fsParams.getFragmentSpace();
539
540 mnt.increase(CounterID.MUTATTEMPTS);
541 mnt.increase(CounterID.NEWCANDIDATEATTEMPTS);
542
543 int numatt = 0;
544 Candidate parent = null;
545 while (numatt < settings.getMaxGeneticOpAttempts())
546 {
547 parent = EAUtils.selectBasedOnFitness(eligibleParents,1, settings)[0];
548 if (parent == null)
549 {
550 numatt++;
551 continue;
552 }
553 break;
554 }
555 mnt.increaseBy(CounterID.MUTPARENTSEARCH,numatt);
556 if (parent == null)
557 {
558 mnt.increase(CounterID.FAILEDMUTATTEMTS);
559 return null;
560 }
561
562 DGraph graph = parent.getGraph().clone();
563 graph.renumberGraphVertices();
564
565 String parentMolName = FilenameUtils.getBaseName(parent.getSDFFile());
566 int parentGraphId = parent.getGraph().getGraphId();
567 graph.setLocalMsg("Mutation:"
568 + " Gen:" + parent.getGeneration() + " Cand:" + parentMolName
569 + "|" + parentGraphId);
570
571 if (!GraphOperations.performMutation(graph,mnt,settings))
572 {
573 mnt.increase(CounterID.FAILEDMUTATTEMTS_PERFORM);
574 mnt.increase(CounterID.FAILEDMUTATTEMTS);
575 return null;
576 }
577
578 graph.setGraphId(GraphUtils.getUniqueGraphIndex());
579
580 graph.addCappingGroups(fragSpace);
581
582 Object[] res = null;
583 try
584 {
585 res = graph.checkConsistency(settings);
586 } catch (NullPointerException|IllegalArgumentException e)
587 {
588 if (!settings.mutatedGraphFailedEvalTolerant)
589 {
590 settings.getLogger().log(Level.INFO, "WRITING DEBUG FILE for "
591 + graph.getLocalMsg());
592 DenoptimIO.writeGraphToSDF(new File("debug_evalGrp_parent.sdf"),
593 parent.getGraph(),false, settings.getLogger(),
594 settings.getRandomizer());
595 DenoptimIO.writeGraphToSDF(new File("debug_evalGrp_curr.sdf"),
596 graph,false, settings.getLogger(),
597 settings.getRandomizer());
598 throw e;
599 } else {
600 res = null;
601 mnt.increase(CounterID.FAILEDMUTATTEMTS_EVAL);
602 }
603 }
604
605 if (res != null)
606 {
607 if (!EAUtils.setupRings(res,graph,settings))
608 {
609 res = null;
610 mnt.increase(CounterID.FAILEDMUTATTEMTS_SETUPRINGS);
611 }
612 } else {
613 mnt.increase(CounterID.FAILEDMUTATTEMTS_EVAL);
614 }
615
616 // Check if the chosen combination gives rise to forbidden ends
617 //TODO this should be considered already when making the list of
618 // possible combination of rings
619 for (Vertex rcv : graph.getFreeRCVertices())
620 {
621 APClass apc = rcv.getEdgeToParent().getSrcAP().getAPClass();
622 if (fragSpace.getCappingMap().get(apc)==null
623 && fragSpace.getForbiddenEndList().contains(apc))
624 {
625 res = null;
626 mnt.increase(CounterID.FAILEDMUTATTEMTS_FORBENDS);
627 }
628 }
629
630 if (res == null)
631 {
632 graph.cleanup();
633 graph = null;
634 mnt.increase(CounterID.FAILEDMUTATTEMTS);
635 return null;
636 }
637
638 Candidate offspring = new Candidate(graph);
639 offspring.setUID(res[0].toString().trim());
640 offspring.setSmiles(res[1].toString().trim());
641 offspring.setChemicalRepresentation((IAtomContainer) res[2]);
642 offspring.setName("M" + GeneralUtils.getPaddedString(
643 DENOPTIMConstants.MOLDIGITS,
644 GraphUtils.getUniqueMoleculeIndex()));
645
646 return offspring;
647 }

715

718 settings.getLogger().log(Level.INFO, msg);
719
720 return candidate;
721 }

784 if (fragSpace.getCappingMap().get(apc)==null

787 res = null;
788 mnt.increase(CounterID.FAILEDBUILDATTEMPTS_FORBIDENDS);
789 }
790 }
791
792 if (res == null)
793 {
794 graph.cleanup();
795 mnt.increase(CounterID.FAILEDBUILDATTEMPTS);
796 return null;
797 }
798

800 candidate.setUID(res[0].toString().trim());
801 candidate.setSmiles(res[1].toString().trim());
802 candidate.setChemicalRepresentation((IAtomContainer) res[2]);
803

807
808 return candidate;
809 }

1390 protected static boolean setupRings(Object[] res, DGraph molGraph,
1391 GAParameters settings) throws DENOPTIMException
1392 {
1393 // get settings //TODO: this should happen inside RunTimeParameters
1394 RingClosureParameters rcParams = new RingClosureParameters();
1395 if (settings.containsParameters(ParametersType.RC_PARAMS))
1396 {
1397 rcParams = (RingClosureParameters)settings.getParameters(
1398 ParametersType.RC_PARAMS);
1399 }
1400 FragmentSpaceParameters fsParams = new FragmentSpaceParameters();
1401 if (settings.containsParameters(ParametersType.FS_PARAMS))
1402 {
1403 fsParams = (FragmentSpaceParameters)settings.getParameters(
1404 ParametersType.FS_PARAMS);
1405 }
1406 FragmentSpace fragSpace = fsParams.getFragmentSpace();
1407
1408 if (!fragSpace.useAPclassBasedApproach())
1409 return true;
1410
1411 if (!rcParams.allowRingClosures())
1412 return true;
1413
1414 // get a atoms/bonds molecular representation (no 3D needed)
1415 ThreeDimTreeBuilder t3d = new ThreeDimTreeBuilder(settings.getLogger(),
1416 settings.getRandomizer());
1417 t3d.setAlignBBsIn3D(false);
1418 IAtomContainer mol = t3d.convertGraphTo3DAtomContainer(molGraph,true);
1419
1420 // Set rotatability property as property of IBond
1421 String rotoSpaceFile = "";
1422 if (settings.containsParameters(ParametersType.FS_PARAMS))
1423 {
1424 rotoSpaceFile = ((FragmentSpaceParameters) settings.getParameters(
1425 ParametersType.FS_PARAMS)).getRotSpaceDefFile();
1426 }
1427 RotationalSpaceUtils.defineRotatableBonds(mol, rotoSpaceFile, true,
1428 true, settings.getLogger());
1429
1430 // get the set of possible RCA combinations = ring closures
1431 CyclicGraphHandler cgh = new CyclicGraphHandler(rcParams,fragSpace);
1432
1433 //TODO: remove hard-coded variable that exclude considering all
1434 // combination of rings
1435 boolean onlyRandomCombOfRings = true;
1436
1437 if (onlyRandomCombOfRings)
1438 {
1439 List<Ring> combsOfRings = cgh.getRandomCombinationOfRings(
1440 mol, molGraph, rcParams.getMaxRingClosures());
1441 if (combsOfRings.size() > 0)
1442 {
1443 for (Ring ring : combsOfRings)
1444 {
1445 // Consider the crowding probability
1446 double shot = settings.getRandomizer().nextDouble();
1447 int crowdOnH = EAUtils.getCrowdedness(
1448 ring.getHeadVertex().getEdgeToParent().getSrcAP(),
1449 true);
1450 int crowdOnT = EAUtils.getCrowdedness(
1451 ring.getTailVertex().getEdgeToParent().getSrcAP(),
1452 true);
1453 double crowdProbH = EAUtils.getCrowdingProbability(crowdOnH,
1454 settings);
1455 double crowdProbT = EAUtils.getCrowdingProbability(crowdOnT,
1456 settings);
1457
1458 if (shot < crowdProbH && shot < crowdProbT)
1459 {
1460 molGraph.addRing(ring);
1461 }
1462 }
1463 }
1464 }
1465 else
1466 {
1467 ArrayList<List<Ring>> allCombsOfRings =
1468 cgh.getPossibleCombinationOfRings(mol, molGraph);
1469
1470 // Keep closable chains that are relevant for chelate formation
1471 if (rcParams.buildChelatesMode())
1472 {
1473 ArrayList<List<Ring>> toRemove = new ArrayList<>();
1474 for (List<Ring> setRings : allCombsOfRings)
1475 {
1476 if (!cgh.checkChelatesGraph(molGraph,setRings))
1477 {
1478 toRemove.add(setRings);
1479 }
1480 }
1481
1482 allCombsOfRings.removeAll(toRemove);
1483 if (allCombsOfRings.isEmpty())
1484 {
1485 String msg = "Setup Rings: no combination of rings.";
1486 settings.getLogger().log(Level.INFO, msg);
1487 return false;
1488 }
1489 }
1490
1491 // Select a combination, if any still available
1492 int sz = allCombsOfRings.size();
1493 if (sz > 0)
1494 {
1495 List<Ring> selected = new ArrayList<>();
1496 if (sz == 1)
1497 {
1498 selected = allCombsOfRings.get(0);
1499 }
1500 else
1501 {
1502 int selId = settings.getRandomizer().nextInt(sz);
1503 selected = allCombsOfRings.get(selId);
1504 }
1505
1506 // append new rings to existing list of rings in graph
1507 for (Ring ring : selected)
1508 {
1509 molGraph.addRing(ring);
1510 }
1511 }
1512 }
1513
1514 // Update the IAtomContainer representation
1515 //DENOPTIMMoleculeUtils.removeUsedRCA(mol,molGraph);
1516 // Done already at t3d.convertGraphTo3DAtomContainer
1517 if (res!=null)
1518 {
1519 res[2] = mol;
1520 }
1521 // Update the SMILES representation
1522 if (res!=null)
1523 {
1524 String molsmiles = MoleculeUtils.getSMILESForMolecule(mol,
1525 settings.getLogger());
1526 if (molsmiles == null)
1527 {
1528 String msg = "Evaluation of graph: SMILES is null! "
1529 + molGraph.toString();
1530 settings.getLogger().log(Level.INFO, msg);
1531 molsmiles = "FAIL: NO SMILES GENERATED";
1532 }
1533 res[1] = molsmiles;
1534 }
1535
1536 // Update the INCHI key representation
1537 if (res!=null)
1538 {
1539 String inchikey = MoleculeUtils.getInChIKeyForMolecule(mol,
1540 settings.getLogger());
1541 if (inchikey == null)
1542 {
1543 String msg = "Evaluation of graph: INCHI is null!";
1544 settings.getLogger().log(Level.INFO, msg);
1545 inchikey = "UNDEFINED";
1546 }
1547 res[0] = inchikey;
1548 }
1549
1550 return true;
1551 }

1691 public static double getProbability(double value,
1692 int scheme, double lambda, double sigmaOne, double sigmaTwo)
1693 {
1694 double prob = 1.0;
1695 if (scheme == 0)
1696 {
1697 double f = Math.exp(-1.0 * value * lambda);
1698 prob = 1 - ((1-f)/(1+f));
1699 }
1700 else if (scheme == 1)
1701 {
1702 prob = 1.0 - Math.tanh(lambda * value);
1703 }
1704 else if (scheme == 2)
1705 {
1706 prob = 1.0-1.0/(1.0 + Math.exp(-sigmaOne * (value - sigmaTwo)));
1707 }
1708 else if (scheme == 3)
1709 {
1710 prob = 1.0;
1711 }
1712 return prob;
