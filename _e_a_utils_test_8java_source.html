85 /* Compatibility matrix
86 *
87 * | A | B | C |
88 * -------------------------
89 * A | T | | |
90 * -------------------------
91 * B | | T | T |
92 * -------------------------
93 * C | | T | T |
94 * -------------------------

120 EmptyVertex rcv = new EmptyVertex();
121 rcv.setBuildingBlockType(BBType.FRAGMENT);
122 rcv.addAP(APCC);
123 rcv.setAsRCV(true);
124 fs.appendVertexToLibrary(rcv, BBType.FRAGMENT, fs.getFragmentLibrary());
125
126 DGraph graphForTemplate = new DGraph();
127 Vertex vg1 = Vertex.newVertexFromLibrary(0,

148 template.setInnerGraph(graphForTemplate);
149 template.setContractLevel(ContractLevel.FIXED_STRUCT);
150 fs.appendVertexToLibrary(template, BBType.SCAFFOLD,
151 fs.getScaffoldLibrary());
152
153 return fsp;

414 @Test
415 public void testBuildByXOver_Embedded_FixedStructure() throws Exception
416 {
417 GAParameters gaparams = PopulationTest.prepare();
418 Population population = new Population(gaparams);
419
420 DGraph[] pair = PopulationTest.getPairOfTestGraphsB();
421 DGraph gA = pair[0];
422 DGraph gB = pair[1];
423 Template embeddedTmplA = (Template) gA.getVertexAtPosition(1);
424 embeddedTmplA.setContractLevel(ContractLevel.FIXED_STRUCT);
425 DGraph embeddedGraphA = embeddedTmplA.getInnerGraph();
426 Template embeddedTmplB = (Template) gB.getVertexAtPosition(1);
427 embeddedTmplB.setContractLevel(ContractLevel.FIXED_STRUCT);
428 DGraph embeddedGraphB = embeddedTmplB.getInnerGraph();
429
430 // Make vertexes unique so, even though they are empty, they will be
431 // seen as non equal and crossover will be seen as non-redundant.
432 String propName = "uniquefier";
433 int i=0;
434 for (Vertex v : embeddedGraphA.getVertexList())
435 {
436 v.setUniquefyingProperty(propName);
437 v.setProperty(propName, i);
438 i++;
439 }
440
441 Candidate cA = new Candidate("CA",gA);
442 population.add(cA);
443 Candidate cB = new Candidate("CB",gB);
444 population.add(cB);
445 ArrayList<Candidate> eligibleParents = new ArrayList<Candidate>();
446 eligibleParents.add(cA);
447 eligibleParents.add(cB);
448
449 Monitor mnt = new Monitor();
450 boolean embeddedGraphHasBeenAlteredA = false;
451 boolean embeddedGraphHasBeenAlteredB = false;
452 for (int ixo=0; ixo<11; ixo++)
453 {
454 Candidate offspring0 = null;
455 Candidate offspring1 = null;
456 try
457 {
458 offspring0 = EAUtils.buildCandidateByXOver(eligibleParents,
459 population, mnt, new int[]{0,1}, ixo, 0, gaparams);
460 offspring1 = EAUtils.buildCandidateByXOver(eligibleParents,
461 population, mnt, new int[]{0,1}, ixo, 1, gaparams);
462 } catch (IndexOutOfBoundsException e)
463 {
464 if (e.getMessage().contains("Index 10 out of bounds"))
465 {
466 // All good! We intentionally triggered this exception to
467 // verify that the list of xover points has the right size.
468 break;
469 }
470 throw e;
471 }
472
473 DGraph g0xo = offspring0.getGraph();
474 DGraph g1xo = offspring1.getGraph();
475 Template t0 = null;
476 Template t1 = null;
477 for (Vertex v : g0xo.getVertexList())
478 {
479 if (v instanceof Template)
480 {
481 t0 = (Template) v;
482 break;
483 }
484 }
485 for (Vertex v : g1xo.getVertexList())
486 {
487 if (v instanceof Template)
488 {
489 t1 = (Template) v;
490 break;
491 }
492 }
493 assertNotNull(t0);
494 assertNotNull(t1);
495
496 DGraph embeddedGraph0 = t0.getInnerGraph();
497 DGraph embeddedGraph1 = t1.getInnerGraph();
498
499 // Ensure there has been a change
500 if (!embeddedGraphA.isIsomorphicTo(embeddedGraph0)
501 && !embeddedGraphA.isIsomorphicTo(embeddedGraph1))
502 {
503 embeddedGraphHasBeenAlteredA = true;
504 }
505 if (!embeddedGraphB.isIsomorphicTo(embeddedGraph0)
506 && !embeddedGraphB.isIsomorphicTo(embeddedGraph1))
507 {
508 embeddedGraphHasBeenAlteredB = true;
509 }
510
511 // Ensure consistency with "fixed-structure" contract
512 assertTrue(embeddedGraphA.isIsostructuralTo(embeddedGraph0)
513 || embeddedGraphB.isIsostructuralTo(embeddedGraph0));
514 assertTrue(embeddedGraphA.isIsostructuralTo(embeddedGraph1)
515 || embeddedGraphB.isIsostructuralTo(embeddedGraph1));
516 }
517 assertTrue(embeddedGraphHasBeenAlteredA);
518 assertTrue(embeddedGraphHasBeenAlteredB);
519 }

523 @Test
524 public void testCandidateGenerationMethod() throws Exception
525 {
526 int ix = 0, im=0, ic=0, tot=1000;
527 double wx = 2, wm = 0.6, wc=0.05;
528 double wtot = wx + wm + wc;
529 Randomizer rng = new Randomizer();
530 for (int i=0; i<tot; i++)
531 {
532 CandidateSource mode = EAUtils.pickNewCandidateGenerationMode(
533 wx, wm, wc, rng);
534 switch (mode)
535 {
536 case CROSSOVER:
537 ix++;
538 break;
539 case MUTATION:
540 im++;
541 break;
542 case CONSTRUCTION:
543 ic++;
544 break;
545 default:
546 assertTrue(false,"Unexpected generation mode "+mode);
547 break;
548 }
549 }
550 double x = ((double)ix) / tot;
551 double m = ((double)im) / tot;
552 double c = ((double)ic) / tot;
553
554 double thld = 0.05;
555
556 assertTrue(Math.abs(x-(wx/wtot)) < thld, "#Xover cases are off!");
557 assertTrue(Math.abs(m-(wm/wtot)) < thld, "#Mutation cases are off!");
558 assertTrue(Math.abs(c-(wc/wtot)) < thld, "#Built cases are off!");
559 }
