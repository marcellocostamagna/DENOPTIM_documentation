255 @Override

257 {
258 EmptyVertex c = new EmptyVertex(getVertexId());
259 c.setAsRCV(this.isRCV());
260 c.setBuildingBlockId(this.getBuildingBlockId());
261 c.setBuildingBlockType(this.getBuildingBlockType());
262

264

271 for (SymmetricSet ss : lstSymAPs)

273 cLstSymAPs.add(ss.clone());

275 c.setSymmetricAPSets(cLstSymAPs);
276 c.setProperties(this.copyStringBasedProperties());
277 if (uniquefyingPropertyKeys!=null)
278 c.uniquefyingPropertyKeys.addAll(uniquefyingPropertyKeys);
279 return c;

318 public boolean sameAs(EmptyVertex other, StringBuilder reason)
319 {
320 if (this.uniquefyingPropertyKeys.size()!=0
321 || other.uniquefyingPropertyKeys.size()!=0)
322 {
323 if (!this.uniquefyingPropertyKeys.equals(other.uniquefyingPropertyKeys))
324 return false;
325
326 for (String k : this.uniquefyingPropertyKeys)
327 {
328 if (!this.properties.get(k).equals(other.properties.get(k)))
329 return false;
330 }
331 }
332 return sameVertexFeatures(other, reason);
333 }
