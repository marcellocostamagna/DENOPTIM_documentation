89 public void interpretKeyword(String key, String value)
90 throws DENOPTIMException
91 {
92 String msg = "";
93 switch (key.toUpperCase())
94 {
95 case "WORKDIR=":
96 workDir = value;
97 break;
98 case "INPUT=":
99 inpFile = new File(value);
100 break;
101 case "OUTPUT=":
102 outFile = new File(value);
103 break;
104 case "EXTRACTTEMPLATES":
105 addTemplatesToLibraries = true;
106 break;
107 case "VERBOSITY=":
108 try
109 {
110 verbosity = Integer.parseInt(value);
111 }
112 catch (Throwable t)
113 {
114 msg = "Unable to understand value " + key + "'" + value + "'";
115 throw new DENOPTIMException(msg);
116 }
117 break;
118 default:
119 msg = "Keyword " + key + " is not a known "
120 + "related keyword for FitnessRunner. Check input files.";
121 throw new DENOPTIMException(msg);
122 }
123 }

132 public void checkParameters() throws DENOPTIMException
133 {
134 String msg = "";
135 if (!workDir.equals(".") && !FileUtils.checkExists(workDir))
136 {
137 msg = "Directory '" + workDir + "' not found. Please specify an "
138 + "existing directory.";
139 throw new DENOPTIMException(msg);
140 }
141
142 if (inpFile != null && !inpFile.exists())
143 {
144 msg = "File with input data not found. Check " + inpFile;
145 throw new DENOPTIMException(msg);
146 }
147
148 if (outFile != null && outFile.exists())
149 {
150 msg = "File meant for output ("
151 + outFile.getAbsolutePath()
152 + ")already exists and we do not overwrite.";
153 throw new DENOPTIMException(msg);
154 }
155 checkOtherParameters();
156 }
157
158//----------------------------------------------------------------------------
159
165 public void processParameters() throws DENOPTIMException
166 {
167 FileUtils.addToRecentFiles(outFile, FileFormat.GRAPHSDF);
168 processOtherParameters();
169
170 StaticLogger.appLogger.log(Level.INFO, "Output files associated with "
171 + "the current run are located in " + workDir);
172 }

182 public String getPrintedList()
183 {
184 StringBuilder sb = new StringBuilder(1024);
185 sb.append(" " + paramTypeName() + " ").append(NL);
186 for (Field f : this.getClass().getDeclaredFields())
187 {
188 try
189 {
190 sb.append(f.getName()).append(" = ").append(
191 f.get(this)).append(NL);
192 }
193 catch (Throwable t)
194 {
195 sb.append("ERROR! Unable to print " + paramTypeName()
196 + " parameters. Cause: " + t);
197 break;
198 }
199 }
200 for (RunTimeParameters otherCollector : otherParameters.values())
201 {
202 sb.append(otherCollector.getPrintedList());
203 }
204 return sb.toString();
205 }
