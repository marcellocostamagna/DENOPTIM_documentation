202 private static ArrayList<SymmetricSet> identifySymmetryRelatedAPSets(
203 IAtomContainer mol,
204 ArrayList<AttachmentPoint> daps)
205 {
206 ArrayList<SymmetricSet> lstCompatible = new ArrayList<>();
207 for (int i=0; i<daps.size()-1; i++)
208 {
209 ArrayList<Integer> lst = new ArrayList<>();
210 Integer i1 = i;
211 lst.add(i1);
212
213 boolean alreadyFound = false;
214 for (SymmetricSet previousSS : lstCompatible)
215 {
216 if (previousSS.contains(i1))
217 {
218 alreadyFound = true;
219 break;
220 }
221 }
222 if (alreadyFound)
223 {
224 continue;
225 }
226
227 AttachmentPoint d1 = daps.get(i);
228 for (int j=i+1; j<daps.size(); j++)
229 {
230 AttachmentPoint d2 = daps.get(j);
231 if (atomsAreCompatible(mol, d1.getAtomPositionNumber(),
232 d2.getAtomPositionNumber()))
233 {
234 // check if reactions are compatible w.r.t. symmetry
235 if (d1.getAPClass()!=null && d2.getAPClass()!=null
236 && d1.getAPClass().compareTo(d2.getAPClass()) == 0)
237 {
238 Integer i2 = j;
239 lst.add(i2);
240 }
241 }
242 }
243
244 if (lst.size() > 1)
245 {
246 lstCompatible.add(new SymmetricSet(lst));
247 }
248 }
249
250 return lstCompatible;
251 }

264 private static boolean atomsAreCompatible(IAtomContainer mol, int a1, int a2)
265 {
266 // check atom types
267 IAtom atm1 = mol.getAtom(a1);
268 IAtom atm2 = mol.getAtom(a2);
269
270 if (atm1.getSymbol().compareTo(atm2.getSymbol()) != 0)
271 return false;
272
273 // check connected bonds
274 if (mol.getConnectedBondsCount(atm1)!=mol.getConnectedBondsCount(atm2))
275 return false;
276
277 // check connected atoms
278 if (mol.getConnectedBondsCount(atm1)!=mol.getConnectedBondsCount(atm2))
279 return false;
280
281 List<IAtom> la1 = mol.getConnectedAtomsList(atm2);
282 List<IAtom> la2 = mol.getConnectedAtomsList(atm2);
283
284 int k = 0;
285 for (int i=0; i<la1.size(); i++)
286 {
287 IAtom b1 = la1.get(i);
288 for (int j=0; j<la2.size(); j++)
289 {
290 IAtom b2 = la2.get(j);
291 if (b1.getSymbol().compareTo(b2.getSymbol()) == 0)
292 {
293 k++;
294 break;
295 }
296 }
297 }
298
299 return k == la1.size();
300 }

552//-----------------------------------------------------------------------------
553
563 public void projectPropertyToAP(String allAtomsProp) throws DENOPTIMException
564 {
565 lstAPs.clear();
566 if (allAtomsProp.trim().equals(""))
567 {
568 return;
569 }
570
571 // Cleanup current APs in atom objects
572 for (int ii=0 ; ii<mol.getAtomCount(); ii++)
573 {
574 IAtom atm = mol.getAtom(ii);
575 atm.removeProperty(DENOPTIMConstants.ATMPROPAPS);
576 }
577
578 // Collect all the APs as objects
579 String[] atomsProp = allAtomsProp.split(
580 DENOPTIMConstants.SEPARATORAPPROPATMS);
581 for (int i = 0; i< atomsProp.length; i++)
582 {
583 String onThisAtm = atomsProp[i];
584 if (onThisAtm.contains(DENOPTIMConstants.SEPARATORAPPROPAPS))
585 {
586 String[] moreAPonThisAtm = onThisAtm.split(
587 DENOPTIMConstants.SEPARATORAPPROPAPS);
588
589 AttachmentPoint ap =
590 new AttachmentPoint(this, moreAPonThisAtm[0]);
591
592 int atmID = ap.getAtomPositionNumber();
593 //WARNING the atmID is already 0-based
594 lstAPs.add(ap);
595 for (int j = 1; j<moreAPonThisAtm.length; j++ )
596 {
597 //WARNING here we have to switch to 1-based enumeration
598 // because we import from SDF string
599 AttachmentPoint apMany =
600 new AttachmentPoint(this, atmID+1
601 + DENOPTIMConstants.SEPARATORAPPROPAAP
602 + moreAPonThisAtm[j]);
603 lstAPs.add(apMany);
604 }
605 }
606 else
607 {
608 AttachmentPoint ap =
609 new AttachmentPoint(this, onThisAtm);
610 lstAPs.add(ap);
611 }
612 }
613
614 projectListAPToAtomProperties();
615
616 updateSymmetryRelations();
617 }

628 public void projectListAPToAtomProperties()

630 // Write attachment points in the atoms
631 for (int i = 0; i < lstAPs.size(); i++)

642 atm.setProperty(DENOPTIMConstants.ATMPROPAPS,oldAPs);

646 ArrayList<AttachmentPoint> aps =
647 new ArrayList<AttachmentPoint>();
648 aps.add(ap);

662 public void projectAPsToProperties()
663 {
664
665 // Prepare the string-representation of unused APs on this graph
666 LinkedHashMap<Integer,List<AttachmentPoint>> apsPerAtom =
667 new LinkedHashMap<>();
668 for (IAtom atm : mol.atoms())
669 {
670 if (atm.getProperty(DENOPTIMConstants.ATMPROPAPS) == null)
671 {
672 continue;
673 }
674 int atmID = mol.indexOf(atm);
675 ArrayList<AttachmentPoint> apsOnAtm =
676 getAPsFromAtom(atm);
677 for (AttachmentPoint ap : apsOnAtm)
678 {
679 if (apsPerAtom.containsKey(atmID))
680 {
681 apsPerAtom.get(atmID).add(ap);
682 } else {
683 List<AttachmentPoint> lst =
684 new ArrayList<AttachmentPoint>();
685 lst.add(ap);
686 apsPerAtom.put(atmID,lst);
687 }
688 }
689 }
690 //WARNING! In the mol.property we use 1-to-n+1 instead of 0-to-n
691 setProperty(DENOPTIMConstants.APSTAG,
692 AttachmentPoint.getAPDefinitionsForSDF(apsPerAtom));
693 }

702 @Override
703 public Fragment clone()
704 {
705 Fragment clone = new Fragment();
706 clone.setVertexId(this.getVertexId());
707 try
708 {
709 clone.mol = MoleculeUtils.makeSameAs(mol);
710 } catch (DENOPTIMException e1)
711 {
712 // TODO Auto-generated catch block
713 e1.printStackTrace();
714 }
715
716 for (AttachmentPoint ap : lstAPs)
717 {
718 AttachmentPoint cAp = new AttachmentPoint(clone,
719 ap.getAtomPositionNumber(),
720 ap.getDirectionVector(),
721 ap.getAPClass());
722 clone.lstAPs.add(cAp);
723 }
724 clone.projectListAPToAtomProperties();
725
726 clone.setBuildingBlockId(this.getBuildingBlockId());
727 clone.setBuildingBlockType(this.getBuildingBlockType());
728
729 clone.setMutationTypes(this.getUnfilteredMutationTypes());
730
731 ArrayList<SymmetricSet> cLstSymAPs = new ArrayList<SymmetricSet>();
732 for (SymmetricSet ss : this.getSymmetricAPSets())
733 {
734 cLstSymAPs.add(ss.clone());
735 }
736 clone.setSymmetricAPSets(cLstSymAPs);
737 clone.setAsRCV(this.isRCV());
738 clone.setProperties(this.copyStringBasedProperties());
739 if (uniquefyingPropertyKeys!=null)
740 clone.uniquefyingPropertyKeys.addAll(uniquefyingPropertyKeys);
741 return clone;
742 }

926 public boolean sameAs(Fragment other, StringBuilder reason)
927 {
928 if (this.containsAtoms() && other.containsAtoms())
929 {
930 if (this.mol.getAtomCount() != other.mol.getAtomCount())
931 {
932 reason.append("Different atom count (" + this.mol.getAtomCount()
933 + ":" + other.mol.getAtomCount() + "); ");
934 }
935 if (this.mol.getBondCount() != other.mol.getBondCount())
936 {
937 reason.append("Different bond count (" + this.mol.getBondCount()
938 + ":" + other.mol.getBondCount() + "); ");
939 }
940 }
941 return sameVertexFeatures(other, reason);

954 public DefaultUndirectedGraph<FragIsomorphNode,FragIsomorphEdge>
955 getJGraphFragIsomorphism()
956 {
957 if (jGraphFragIsomorphism != null)
958 {
959 return jGraphFragIsomorphism;
960 }
961 jGraphFragIsomorphism = new DefaultUndirectedGraph<>(
962 FragIsomorphEdge.class);
963
964 Map<IAtom,FragIsomorphNode> atmToNode =
965 new HashMap<IAtom,FragIsomorphNode>();
966 Set<IAtom> ignoredDu = new HashSet<IAtom>();
967 for (IAtom atm : mol.atoms())
968 {
969 if (DENOPTIMConstants.DUMMYATMSYMBOL.equals(
970 MoleculeUtils.getSymbolOrLabel(atm))
971 && mol.getConnectedBondsCount(atm)<2)
972 {
973 //Ignore Du on linearities
974 ignoredDu.add(atm);
975 continue;
976 }
977 FragIsomorphNode node = new FragIsomorphNode(atm);
978 atmToNode.put(atm,node);
979 jGraphFragIsomorphism.addVertex(node);
980 }
981 for (IBond bnd : mol.bonds())
982 {
983 if (bnd.getAtomCount() > 2)
984 {
985 throw new IllegalArgumentException("Cannot handle bonds that involve "
986 + "more than two atoms");
987 }
988 if (ignoredDu.contains(bnd.getAtom(0))
989 || ignoredDu.contains(bnd.getAtom(0)))
990 {
991 //Ignore Du on linearities
992 continue;
993 }
994
995 FragIsomorphEdge edge = new FragIsomorphEdge(bnd);
996 jGraphFragIsomorphism.addEdge(atmToNode.get(bnd.getAtom(0)),
997 atmToNode.get(bnd.getAtom(1)), edge);
998 }
999 for (AttachmentPoint ap : getAttachmentPoints())
1000 {
1001 FragIsomorphNode node = new FragIsomorphNode(ap);
1002 jGraphFragIsomorphism.addVertex(node);
1003 FragIsomorphEdge edge = new FragIsomorphEdge();
1004 jGraphFragIsomorphism.addEdge(atmToNode.get(mol.getAtom(
1005 ap.getAtomPositionNumber())), node, edge);
1006 }
1007
1008 return jGraphFragIsomorphism;
1009 }

1025 public boolean isIsomorphicTo(Vertex other)
1026 {
1027 if (!(other instanceof Fragment))
1028 {
1029 return false;
1030 }
1031 Fragment otherFrag = (Fragment) other;
1032
1033 // NB: do not add a comparison of #atoms or #bonds because they may be
1034 // different as a result of a different number of dummy atoms placed
1035 // on linearities. Such atoms are not considered when checking for
1036 // isomorphism so, de facto, fragments can be isomorphic and have a
1037 // different number of atoms and bonds.
1038
1039 if (this.getAttachmentPoints().size()
1040 != otherFrag.getAttachmentPoints().size())
1041 {
1042 return false;
1043 }
1044
