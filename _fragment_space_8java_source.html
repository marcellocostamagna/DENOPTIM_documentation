521 public Vertex getVertexFromLibrary(
522 Vertex.BBType bbType, int bbIdx) throws DENOPTIMException
523 {
524 String msg = "";
525 switch (bbType)
526 {
527 case SCAFFOLD:
528 if (scaffoldLib == null)
529 {
530 msg = "Cannot retrieve scaffolds before initialising the "
531 + "scaffold library.";
532 throw new DENOPTIMException(msg);
533 }
534 break;
535 case FRAGMENT:
536 if (fragmentLib == null)
537 {
538 msg = "Cannot retrieve fragments before initialising the "
539 + "fragment library.";
540 throw new DENOPTIMException(msg);
541 }
542 break;
543 case CAP:
544 if (cappingLib == null)
545 {
546 msg = "Cannot retrieve capping groups before initialising"
547 + "the library of capping groups.";
548 throw new DENOPTIMException(msg);
549 }
550 break;
551
552 default:
553 throw new DENOPTIMException("Cannot find building block of "
554 + "type '" + bbType + "' in the fragment space.");
555 }
556
557 Vertex originalVrtx = null;
558 switch (bbType)
559 {
560 case SCAFFOLD:
561 if (bbIdx >-1 && bbIdx < scaffoldLib.size())
562 {
563 originalVrtx = scaffoldLib.get(bbIdx);
564 }
565 else
566 {
567 msg = "Mismatch between scaffold bbIdx (" + bbIdx
568 + ") and size of the library (" + scaffoldLib.size()
569 + "). FragType: " + bbType;
570 settings.getLogger().log(Level.SEVERE, msg);
571 throw new DENOPTIMException(msg);
572 }
573 break;
574
575 case FRAGMENT:
576 if (bbIdx >-1 && bbIdx < fragmentLib.size())
577 {
578 originalVrtx = fragmentLib.get(bbIdx);
579 }
580 else
581 {
582 msg = "Mismatch between fragment bbIdx (" + bbIdx
583 + ") and size of the library (" + fragmentLib.size()
584 + "). FragType: " + bbType;
585 settings.getLogger().log(Level.SEVERE, msg);
586 throw new DENOPTIMException(msg);
587 }
588 break;
589
590 case CAP:
591 if (bbIdx >-1 && bbIdx < cappingLib.size())
592 {
593 originalVrtx = cappingLib.get(bbIdx);
594 }
595 else
596 {
597 msg = "Mismatch between capping group bbIdx " + bbIdx
598 + ") and size of the library (" + cappingLib.size()
599 + "). FragType: " + bbType;
600 settings.getLogger().log(Level.SEVERE, msg);
601 throw new DENOPTIMException(msg);
602 }
603 break;
604
605 case UNDEFINED:
606 msg = "Attempting to take UNDEFINED type of building block from "
607 + "fragment library.";
608 settings.getLogger().log(Level.WARNING, msg);
609 if (bbIdx < fragmentLib.size())
610 {
611 originalVrtx = fragmentLib.get(bbIdx);
612 }
613 else
614 {
615 msg = "Mismatch between fragment bbIdx (" + bbIdx
616 + ") and size of the library (" + fragmentLib.size()
617 + "). FragType: " + bbType;
618 settings.getLogger().log(Level.SEVERE, msg);
619 throw new DENOPTIMException(msg);
620 }
621 break;
622
623 default:
624 msg = "Unknown type of fragment '" + bbType + "'.";
625 settings.getLogger().log(Level.SEVERE, msg);
626 throw new DENOPTIMException(msg);
627 }
628 Vertex clone = originalVrtx.clone();
629
630 clone.setVertexId(GraphUtils.getUniqueVertexIndex());
631
632 clone.setBuildingBlockId(bbIdx);
633 if (originalVrtx.getBuildingBlockId() != bbIdx)
634 {
635 settings.getLogger().log(Level.WARNING, "Mismatch between building "
636 + "block ID ("
637 + originalVrtx.getBuildingBlockId() + ") and position in "
638 + "the list of building blocks (" + bbIdx + ") for type "
639 + bbType + ".");
640 }
641 return clone;
642 }
