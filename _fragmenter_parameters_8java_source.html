885 public void interpretKeyword(String key, String value)
886 throws DENOPTIMException
887 {
888 String msg = "";
889 switch (key.toUpperCase())
890 {
891 case "WORKDIR=":
892 workDir = value;
893 break;
894
895 case "STRUCTURESFILE=":
896 structuresFile = value;
897 break;
898
899 case "FORMULATXTFILE=":
900 checkFormula = true;
901 formulaeFile = value;
902 break;
903
904 case "PREFILTERSMARTS=":
905 preFilter = true;
906 preFilterSMARTS.add(value);
907 break;
908
909 case "CUTTINGRULESFILE=":
910 doFragmentation = true;
911 cutRulesFile = value;
912 break;
913
914 case "UNSETTOSINGLEBO":
915 acceptUnsetToSingeBOApprox = true;
916 break;
917
918 case "IGNORABLEFRAGMENTS=":
919 ignorableFragmentsFile = value;
920 doFiltering = true;
921 break;
922
923 case "TARGETFRAGMENTS=":
924 targetFragmentsFile = value;
925 doFiltering = true;
926 break;
927
928 case "ISOMORPHICSAMPLESIZE=":
929 try {
930 isomorphicSampleSize = Integer.parseInt(value);
931 } catch (Throwable t)
932 {
933 msg = "Unable to parse value of " + key + ": '" + value + "'";
934 throw new DENOPTIMException(msg);
935 }
936 if (isomorphicSampleSize>0)
937 doManageIsomorphicFamilies = true;
938 break;
939
940 case "REMOVEDUPLICATES":
941 doManageIsomorphicFamilies = true;
942 break;
943
944 case "MWSLOTSIZE=":
945 try {
946 mwSlotSize = Integer.parseInt(value);
947 } catch (Throwable t)
948 {
949 msg = "Unable to parse value of " + key + ": '" + value + "'";
950 throw new DENOPTIMException(msg);
951 }
952 break;
953
954 case "REJECTMINORISOTOPES":
955 doRejectWeirdIsotopes = true;
956 doFiltering = true;
957 break;
958
959 case "REJECTELEMENT=":
960 rejectedElements.add(value);
961 doFiltering = true;
962 break;
963
964 case "REJFORMULALESSTHAN=":
965 if (formulaCriteriaLessThan.size()>0)
966 {
967 msg = "Attempt to specify more than one criterion for "
968 + "rejecting fragments based on a lower-limit "
969 + "molecular formula. ";
970 throw new DENOPTIMException(msg);
971 }
972 Map<String,Double> elSymbolsCount = null;
973 try {
974 elSymbolsCount = FormulaUtils.parseFormula(value);
975 } catch (Throwable t)
976 {
977 msg = "Unable to parse value of " + key + ": '" + value + "'";
978 throw new DENOPTIMException(msg);
979 }
980 formulaCriteriaLessThan = elSymbolsCount;
981 doFiltering = true;
982 break;
983
984 case "REJFORMULAMORETHAN=":
985 Map<String,Double> elSymbolsCount2= null;
986 try {
987 elSymbolsCount2 = FormulaUtils.parseFormula(value);
988 } catch (Throwable t)
989 {
990 msg = "Unable to parse value of " + key + ": '" + value + "'";
991 throw new DENOPTIMException(msg);
992 }
993 formulaCriteriaMoreThan.add(elSymbolsCount2);
994 doFiltering = true;
995 break;
996
997 case "REJECTAPCLASS=":
998 rejectedAPClasses.add(value);
999 doFiltering = true;
1000 break;
1001
1002 case "REJECTAPCLASSCOMBINATION=":
1003 String[] lst = value.split("\\s+");
1004 rejectedAPClassCombinations.add(lst);
1005 doFiltering = true;
1006 break;
1007
1008 case "MAXFRAGSIZE=":
1009 try {
1010 maxFragHeavyAtomCount = Integer.parseInt(value);
1011 } catch (Throwable t)
1012 {
1013 msg = "Unable to parse value of " + key + ": '" + value + "'";
1014 throw new DENOPTIMException(msg);
1015 }
1016 doFiltering = true;
1017 break;
1018
1019 case "MINFRAGSIZE=":
1020 try {
1021 minFragHeavyAtomCount = Integer.parseInt(value);
1022 } catch (Throwable t)
1023 {
1024 msg = "Unable to parse value of " + key + ": '" + value + "'";
1025 throw new DENOPTIMException(msg);
1026 }
1027 doFiltering = true;
1028 break;
1029
1030 case "REJECTSMARTS=":
1031 fragRejectionSMARTS.put(value, value);
1032 doFiltering = true;
1033 break;
1034
1035 case "RETAINSMARTS=":
1036 fragRetentionSMARTS.put(value, value);
1037 doFiltering = true;
1038 break;
1039
1040 case "CLUSTERIZEANDCOLLECT=":
1041 doManageIsomorphicFamilies = true;
1042 doExtactRepresentativeConformer = true;
1043 switch (value.trim().toUpperCase())
1044 {
1045 case "CENTROIDS":
1046 useCentroidsAsRepresentativeConformer = true;
1047 break;
1048
1049 case "MOSTCENTRAL":
1050 useCentroidsAsRepresentativeConformer = false;
1051 break;
1052
1053 default:
1054 throw new DENOPTIMException("Unable to parse value of "
1055 + key + ": '" + value + "'");
1056 }
1057 break;
1058
1059 case "SAVECLUSTERS":
1060 doExtactRepresentativeConformer = true;
1061 saveClustersOfConformerToFile = true;
1062 break;
1063
1064 case "SIZEUNIMODALPOPULATION=":
1065 sizeUnimodalPop = Integer.parseInt(value);
1066 break;
1067
1068 case "MAXNOISEUNIMODALPOPULATION=":
1069 maxNoiseUnimodalPop = Double.parseDouble(value);
1070 break;
1071
1072 case "SDWEIGHTUNIMODALPOPULATION=":
1073 factorForSDOnStatsOfUnimodalPop = Double.parseDouble(value);
1074 break;
1075
1076/*
1077 case "=":
1078 = value;
1079 doFiltering = true;
1080 break;
1081 */
1082
1083 case "PARALLELTASKS=":
1084 try
1085 {
1086 numParallelTasks = Integer.parseInt(value);
1087 }
1088 catch (Throwable t)
1089 {
1090 msg = "Unable to understand value " + key + "'" + value + "'";
1091 throw new DENOPTIMException(msg);
1092 }
1093 break;
1094
1095 case "VERBOSITY=":
1096 try
1097 {
1098 verbosity = Integer.parseInt(value);
1099 }
1100 catch (Throwable t)
1101 {
1102 msg = "Unable to understand value " + key + "'" + value + "'";
1103 throw new DENOPTIMException(msg);
1104 }
1105 break;
1106
1107 default:
1108 msg = "Keyword " + key + " is not a known Fragmenter-"
1109 + "related keyword. Check input files.";
1110 throw new DENOPTIMException(msg);
1111 }
1112 }

1121 public void checkParameters() throws DENOPTIMException
1122 {
1123 if (!workDir.equals(System.getProperty("user.dir")))
1124 {
1125 ensureFileExists(workDir);
1126 }
1127 ensureIsPositive("numParallelTasks", numParallelTasks, "PARALLELTASKS");
1128 ensureIsPositive("isomorphicSampleSize", isomorphicSampleSize,
1129 "ISOMORPHICSAMPLESIZE");
1130 ensureIsPositive("mwSlotSize", mwSlotSize, "MWSLOTSIZE");
1131 ensureFileExistsIfSet(structuresFile);
1132 ensureFileExistsIfSet(cutRulesFile);
1133 ensureFileExistsIfSet(formulaeFile);
1134 ensureFileExistsIfSet(ignorableFragmentsFile);
1135 ensureFileExistsIfSet(targetFragmentsFile);
1136
1137 checkOtherParameters();
1138 }

1140//------------------------------------------------------------------------------
1141
1147 public void processParameters() throws DENOPTIMException
1148 {
1149 if (isMaster)
1150 createWorkingDirectory();
1151
1152 cuttingRules = new ArrayList<CuttingRule>();
1153 if (cutRulesFile!=null && !cutRulesFile.isBlank())
1154 {
1155 DenoptimIO.readCuttingRules(new File(cutRulesFile), cuttingRules);
1156 }
1157 if (formulaeFile!=null && !formulaeFile.isBlank())
1158 {
1159 formulae = DenoptimIO.readCSDFormulae(new File(formulaeFile));
1160 }
1161 processOtherParameters();
1162
1163 if (ignorableFragmentsFile!=null && !ignorableFragmentsFile.isBlank())
1164 {
1165 try
1166 {
1167 ignorableFragments = DenoptimIO.readVertexes(
1168 new File(ignorableFragmentsFile), BBType.UNDEFINED);
1169 } catch (Throwable e)
1170 {
1171 throw new DENOPTIMException("Problems reading file '"
1172 + ignorableFragmentsFile + "'", e);
1173 }
1174 }
1175
1176 if (targetFragmentsFile!=null && !targetFragmentsFile.isBlank())
1177 {
1178 try
1179 {
1180 targetFragments = DenoptimIO.readVertexes(
1181 new File(targetFragmentsFile), BBType.UNDEFINED);
1182 } catch (Throwable e)
1183 {
1184 throw new DENOPTIMException("Problems reading file '"
1185 + targetFragmentsFile + "'", e);
1186 }
1187 }
1188
1189 if (!doFragmentation && doExtactRepresentativeConformer)
1190 {
1191 isStandaloneFragmentClustering = true;
1192 }
1193
1194 if (isMaster)
1195 {
1196 StaticLogger.appLogger.log(Level.INFO, "Program log file: "
1197 + logFile + DENOPTIMConstants.EOL
1198 + "Output files associated with the current run are "
1199 + "located in " + workDir);
1200 }
1201 }

1203//------------------------------------------------------------------------------
1204
1205 private void createWorkingDirectory()
1206 {
1207 String curDir = workDir;
1208 String fileSep = System.getProperty("file.separator");
1209 boolean success = false;
1210 while (!success)
1211 {
1212 SimpleDateFormat sdf = new SimpleDateFormat("yyyyMMddkkmmss");
1213 String str = "FRG" + sdf.format(new Date());
1214 workDir = curDir + fileSep + str;
1215 success = FileUtils.createDirectory(workDir);
1216 }
1217 FileUtils.addToRecentFiles(workDir, FileFormat.FRG_RUN);
1218 logFile = workDir + ".log";
1219 }

1229 public String getPrintedList()

1243 + " parameters. Cause: " + t);
1244 break;
1245 }
1246 }
1247 for (RunTimeParameters otherCollector : otherParameters.values())
1248 {
