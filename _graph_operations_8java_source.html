83 public static List<XoverSite> locateCompatibleXOverPoints(
84 DGraph graphA, DGraph graphB, FragmentSpace fragSpace)
85 throws DENOPTIMException
86 {
87 // First, we identify all the edges that allow crossover, and collect
88 // their target vertexes (i.e., all the potential seed vertexes of
89 // subgraphs that crossover could swap.
90 List<Vertex[]> compatibleVrtxPairs = new ArrayList<Vertex[]>();
91 for (Edge eA : graphA.getEdgeList())
92 {
93 Vertex vA = eA.getTrgAP().getOwner();
94 // We don't do genetic operations on capping vertexes
95 if (vA.getBuildingBlockType() == BBType.CAP)
96 continue;
97
98 for (Edge eB : graphB.getEdgeList())
99 {
100 Vertex vB = eB.getTrgAP().getOwner();
101 // We don't do genetic operations on capping vertexes
102 if (vB.getBuildingBlockType() == BBType.CAP)
103 continue;
104
105 //Check condition for considering this combination
106 if (isCrossoverPossible(eA, eB, fragSpace))
107 {
108 Vertex[] pair = new Vertex[]{vA,vB};
109 compatibleVrtxPairs.add(pair);
110 }
111 }
112 }
113
114 // The crossover sites are the combination of the above compatible
115 // vertexes that define subgraphs respecting the requirements for
116 // being swapped between the two graphs.
117 ArrayList<XoverSite> sites = new ArrayList<XoverSite>();
118 for (Vertex[] pair : compatibleVrtxPairs)
119 {
120 Vertex vA = pair[0];
121 Vertex vB = pair[1];
122 DGraph gA = vA.getGraphOwner();
123 DGraph gB = vB.getGraphOwner();
124
125 // Here we also identify the branch identity of each descendant
126 List<Vertex> descendantsA = new ArrayList<Vertex>();
127 gA.getChildrenTree(vA, descendantsA, true);
128 List<Vertex> descendantsB = new ArrayList<Vertex>();
129 gB.getChildrenTree(vB, descendantsB, true);
130
131 // Branches that are isomorphic are not considered for crossover
132 DGraph test1 = gA.clone();
133 DGraph test2 = gB.clone();
134 try
135 {
136 DGraph subGraph1 = test1.extractSubgraph(gA.indexOf(vA));
137 DGraph subGraph2 = test2.extractSubgraph(gB.indexOf(vB));
138 if (!subGraph1.isIsomorphicTo(subGraph2))
139 {
140 List<Vertex> branchOnVA = new ArrayList<Vertex>();
141 branchOnVA.add(vA);
142 branchOnVA.addAll(descendantsA);
143 List<Vertex> branchOnVB = new ArrayList<Vertex>();
144 branchOnVB.add(vB);
145 branchOnVB.addAll(descendantsB);
146
147 checkAndAddXoverSites(fragSpace, branchOnVA, branchOnVB,
148 CrossoverType.BRANCH, sites);
149 }
150 } catch (DENOPTIMException e)
151 {
152 //We should never end up here.
153 e.printStackTrace();
154 }
155
156 // To limit the number of combination, we first get rid of end-point
157 // candidates that cannot be used
158 List<Vertex[]> usablePairs = new ArrayList<Vertex[]>();
159
160 // To identify subgraphs smaller than the full branch we need to find
161 // where such subgraphs end, i.e., the vertexes at the end of
162 // such subgraphs (included in them), a.k.a. the subgraph ends.
163 // Since these subgraph ends will need to allow connection with the
164 // rest of the original graph, they are related to the
165 // already-identified crossover-compatible
166 // sites, i.e., they are the parents of the vertexes collected in
167 // compatibleVrtxPairs. Yet, we can combine them
168 // * in any number from 1 to all of them,
169 // * in any combination of the chosen number of them.
170 // Also, note that the ends need not to cover all the branches. So,
171 // some combinations will have to cut some branches short while
172 // taking some other branches completely till their last leaf.
173 for (Vertex[] otherPair : compatibleVrtxPairs)
174 {
175 // NB: the xover compatible sites are the child vertexes of the
176 // subgraph ends. So we need to get the parent
177 Vertex nextToEndOnA = otherPair[0];
178 Vertex nextToEndOnB = otherPair[1];
179 Vertex endOnA = nextToEndOnA.getParent();
180 Vertex endOnB = nextToEndOnB.getParent();
181 if (endOnA==null || endOnB==null)
182 continue;
183
184 // Exclude vertexes that are not downstream to the seed of the subgraph
185 if (!descendantsA.contains(endOnA) && endOnA!=vA
186 || !descendantsB.contains(endOnB) && endOnB!=vB)
187 continue;
188
189 // If any partner is a fixed-structure templates...
190 if ((gA.getTemplateJacket()!=null
191 && gA.getTemplateJacket().getContractLevel()
192 == ContractLevel.FIXED_STRUCT)
193 || (gB.getTemplateJacket()!=null
194 && gB.getTemplateJacket().getContractLevel()
195 == ContractLevel.FIXED_STRUCT))
196 {
197 //...the two paths must have same length. This would be
198 // checked anyway later when checking for isostructural
199 // subgraphs, but here it helps reducing the size of the
200 // combinatorial problem.
201 PathSubGraph pathA = new PathSubGraph(vA, endOnA, gA);
202 PathSubGraph pathB = new PathSubGraph(vB, endOnB, gB);
203 if (pathA.getPathLength()!=pathB.getPathLength())
204 continue;
205 }
206 Vertex[] pairOfEnds = new Vertex[]{endOnA,endOnB};
207 if (!usablePairs.contains(pairOfEnds))
208 usablePairs.add(pairOfEnds);
209 }
210
211 // We classify the pairs by branch ownership
212 TreeMap<String,List<Vertex[]>> sitesByBranchIdA =
213 new TreeMap<String,List<Vertex[]>>();
214 TreeMap<String,List<Vertex[]>> sitesByBranchIdB =
215 new TreeMap<String,List<Vertex[]>>();
216 for (Vertex[] pp : usablePairs)
217 {
218 String branchIdA = gA.getBranchIdOfVertexAsStr(pp[0]);
219 String branchIdB = gB.getBranchIdOfVertexAsStr(pp[1]);
220 if (sitesByBranchIdA.containsKey(branchIdA))
221 {
222 sitesByBranchIdA.get(branchIdA).add(pp);
223 } else {
224 ArrayList<Vertex[]> lst = new ArrayList<Vertex[]>();
225 lst.add(pp);
226 sitesByBranchIdA.put(branchIdA, lst);
227 }
228 if (sitesByBranchIdB.containsKey(branchIdB))
229 {
230 sitesByBranchIdB.get(branchIdB).add(pp);
231 } else {
232 ArrayList<Vertex[]> lst = new ArrayList<Vertex[]>();
233 lst.add(pp);
234 sitesByBranchIdB.put(branchIdB, lst);
235 }
236 }
237
238 // The side with the smallest set of branches determines the max
239 // number of pairs that can define a subgraph.
240 TreeMap<String,List<Vertex[]>> fewestBranchesSide = null;
241 if (sitesByBranchIdA.size() <= sitesByBranchIdB.size())
242 fewestBranchesSide = sitesByBranchIdA;
243 else
244 fewestBranchesSide = sitesByBranchIdB;
245
246 // Add the empty, i.e., the branch with empty is not cut short.
247 for (List<Vertex[]> val : fewestBranchesSide.values())
248 val.add(new Vertex[]{null,null});
249
250 // Generate the combinations: Cartesian product of multiple lists.
251 // NB: this combinatorial generator retains the
252 // sequence of generated subsets (important for reproducibility)
253 List<List<Vertex[]>> preCombsOfEnds = Generator.cartesianProduct(
254 fewestBranchesSide.values())
255 .stream()
256 .collect(Collectors.<List<Vertex[]>>toList());
257
258 // Remove the 'null,null' place holders that indicate the use of no
259 // end-point on a specific branch.
260 List<List<Vertex[]>> combsOfEnds = new ArrayList<List<Vertex[]>>();
261 for (List<Vertex[]> comb : preCombsOfEnds)
262 {
263 List<Vertex[]> nullPurgedComb = new ArrayList<Vertex[]>();
264 for (Vertex[] inPair : comb)
265 {
266 if (inPair[0]!=null && inPair[1]!=null)
267 nullPurgedComb.add(inPair);
268 }
269 // the case where all entries are null corresponds to use no
270 // end point on any branch, which is already considered above.
271 if (nullPurgedComb.size()>0)
272 combsOfEnds.add(nullPurgedComb);
273 }
274
275 // This would be a strategy to parallelize. However, this type of
276 // parallelization is not compatible with ensuring reproducibility.
277 // Perhaps, we could guarantee reproducibility by acting on the
278 // collector of sites as to ensure the list is sorted
279 // at the end of the generation of all possibilities.
280 /*
281 combsOfEnds
282 .parallelStream()
283 .limit(50) // Prevent explosion!
284 .forEach(c -> processCombinationOfEndPoints(pair, c, sites,
285 fragSpace));
286 */
287
288 combsOfEnds.stream()
289 .limit(50) // Prevent explosion!
290 .forEach(c -> processCombinationOfEndPoints(pair, c, sites,
291 fragSpace));
292 }
293
294 // NB: we consider only templates that are at the same level of embedding
295 for (Vertex vA : graphA.getVertexList())
296 {
297 if (!(vA instanceof Template))
298 continue;
299 Template tA = (Template) vA;
300
301 if (tA.getContractLevel() == ContractLevel.FIXED)
302 continue;
303
304 for (Vertex vB : graphB.getVertexList())
305 {
306 if (!(vB instanceof Template))
307 continue;
308 Template tB = (Template) vB;
309
310 if (tB.getContractLevel() == ContractLevel.FIXED)
311 continue;
312
313 for (XoverSite xos : locateCompatibleXOverPoints(
314 tA.getInnerGraph(), tB.getInnerGraph(), fragSpace))
315 {
316 if (!sites.contains(xos))
317 sites.add(xos);
318 }
319 }
320 }
321 return sites;
322 }

338 private static void processCombinationOfEndPoints(Vertex[] pair,
339 List<Vertex[]> cominationOfEnds,
340 List<XoverSite> collector, FragmentSpace fragSpace)
341 {
342 // Empty set corresponds to using the entire branch and subgraph and
343 // has been already dealt with at this point
344 if (cominationOfEnds.size()==0)
345 return;
346
347 // This would be a strategy to parallelize. However, this type of
348 // parallelization is not compatible with ensuring reproducibility.
349 // Perhaps, we could guarantee reproducibility by acting on the
350 // collector as to ensure the list of collected results is sorted
351 // at the end of the generation of all possibilities
352 /*
353 List<List<Vertex[]>> permutsOfEnds = new ArrayList<List<Vertex[]>>();
354 Generator.subset(cominationOfEnds)
355 .simple()
356 .stream()
357 .limit(100) // Prevent explosion!
358 .forEach(c -> permutsOfEnds.add(c));
359 permutsOfEnds
360 .parallelStream()
361 .forEach(c -> processPermutationOfEndPoints(pair, c, collector,
362 fragSpace));
363 */
364
365 Generator.permutation(cominationOfEnds)
366 .simple()
367 .stream()
368 .limit(100) // Prevent explosion!
369 .forEach(c -> processPermutationOfEndPoints(pair, c, collector,
370 fragSpace));
371 }

387 private static void processPermutationOfEndPoints(Vertex[] pair,
388 List<Vertex[]> chosenSequenceOfEndpoints,
389 List<XoverSite> collector, FragmentSpace fragSpace)
390 {
391 Vertex vA = pair[0];
392 Vertex vB = pair[1];
393 DGraph gA = vA.getGraphOwner();
394 DGraph gB = vB.getGraphOwner();
395
396 // Exclude overlapping combinations
397 boolean exclude = false;
398 for (Vertex[] pairA : chosenSequenceOfEndpoints)
399 {
400 for (Vertex[] pairB : chosenSequenceOfEndpoints)
401 {
402 if (pairA==pairB)
403 continue;
404
405 if (pairA[0]==pairB[0] || pairA[1]==pairB[1])
406 {
407 exclude = true;
408 break;
409 }
410 }
411 if (exclude)
412 break;
413 }
414 if (exclude)
415 return;
416
417 List<Vertex> subGraphEndInA = new ArrayList<Vertex>();
418 List<Vertex> subGraphEndInB = new ArrayList<Vertex>();
419 List<Vertex> alreadyIncludedFromA = new ArrayList<Vertex>();
420 List<Vertex> alreadyIncludedFromB = new ArrayList<Vertex>();
421 for (Vertex[] otherPair : chosenSequenceOfEndpoints)
422 {
423 Vertex endOnA = otherPair[0];
424 Vertex endOnB = otherPair[1];
425
426 // Ignore vertexes that are already part of the subgraph
427 if (alreadyIncludedFromA.contains(endOnA)
428 || alreadyIncludedFromB.contains(endOnB))
429 continue;
430
431 PathSubGraph pathA = new PathSubGraph(vA, endOnA, gA);
432 PathSubGraph pathB = new PathSubGraph(vB, endOnB, gB);
433 subGraphEndInA.add(endOnA);
434 subGraphEndInB.add(endOnB);
435 alreadyIncludedFromA.addAll(pathA.getVertecesPath());
436 alreadyIncludedFromB.addAll(pathB.getVertecesPath());
437 }
438 ArrayList<Vertex> subGraphA = new ArrayList<Vertex>();
439 subGraphA.add(vA);
440 if (!subGraphEndInA.contains(vA))
441 gA.getChildTreeLimited(vA, subGraphA, subGraphEndInA, true);
442
443 ArrayList<Vertex> subGraphB = new ArrayList<Vertex>();
444 subGraphB.add(vB);
445 if (!subGraphEndInB.contains(vB))
446 gB.getChildTreeLimited(vB, subGraphB, subGraphEndInB, true);
447
448 // The two subgraphs must not be isomorfic to prevent unproductive crossover
449 if (subGraphA.size()>1 && subGraphB.size()>1)
450 {
451 DGraph subGraphCloneA = gA.extractSubgraph(subGraphA);
452 DGraph subGraphCloneB = gB.extractSubgraph(subGraphB);
453 if (subGraphCloneA.isIsomorphicTo(subGraphCloneB))
454 return;
455 } else {
456 if (subGraphA.get(0).sameAs(subGraphB.get(0), new StringBuilder()))
457 return;
458 }
459
460 checkAndAddXoverSites(fragSpace, subGraphA, subGraphB,
461 CrossoverType.SUBGRAPH, collector);
462 }

475 private static void checkAndAddXoverSites(FragmentSpace fragSpace,
476 List<Vertex> subGraphA,
477 List<Vertex> subGraphB, CrossoverType xoverType,
478 List<XoverSite> collector)
479 {
480 DGraph gOwnerA = subGraphA.get(0).getGraphOwner();
481 DGraph gOwnerB = subGraphB.get(0).getGraphOwner();
482
483 // What APs need to find a corresponding AP in the other
484 // subgraph in order to allow swapping?
485 List<AttachmentPoint> needyAPsA = gOwnerA.getInterfaceAPs(
486 subGraphA);
487 List<AttachmentPoint> allAPsA = gOwnerA.getSubgraphAPs(
488 subGraphA);
489 List<AttachmentPoint> needyAPsB = gOwnerB.getInterfaceAPs(
490 subGraphB);
491 List<AttachmentPoint> allAPsB = gOwnerB.getSubgraphAPs(
492 subGraphB);
493 if (allAPsA.size() < needyAPsB.size()
494 || allAPsB.size() < needyAPsA.size())
495 {
496 // Impossible to satisfy needy APs. Crossover site is not usable.
497 return;
498 }
499
500 // Shortcut: since the compatibility of one AP that is needed to do
501 // branch swapping is guaranteed by the identification of the seeds of
502 // swappable subgraphs, we can avoid searching for a valid A mapping
503 // when the graphs are not embedded. This because any AP that is not
504 // the one connecting the subgraph to the parent can be left free or
505 // removed without side effects on templates that embed the graph
506 // because there is no such template.
507 Template jacketTmplA = gOwnerA.getTemplateJacket();
508 Template jacketTmplB = gOwnerB.getTemplateJacket();
509 if (xoverType == CrossoverType.BRANCH
510 && jacketTmplA==null
511 && jacketTmplB==null)
512 {
513 XoverSite xos = new XoverSite(subGraphA, needyAPsA, subGraphB,
514 needyAPsB, xoverType);
515 if (!collector.contains(xos))
516 collector.add(xos);
517 return;
518 }
519
520 if ((jacketTmplA!=null && jacketTmplA.getContractLevel()
521 ==ContractLevel.FIXED_STRUCT)
522 || (jacketTmplB!=null && jacketTmplB.getContractLevel()
523 ==ContractLevel.FIXED_STRUCT))
524 {
525 // Avoid to alter cyclicity of inner graphs
526 for (Vertex v : subGraphA)
527 if (v.isRCV() && !gOwnerA.getRingsInvolvingVertex(v).isEmpty())
528 return;
529 for (Vertex v : subGraphB)
530 if (v.isRCV() && !gOwnerB.getRingsInvolvingVertex(v).isEmpty())
531 return;
532
533 // Avoid to change the structure of inner graphs
534 DGraph subGraphCloneA = gOwnerA.extractSubgraph(subGraphA);
535 DGraph subGraphCloneB = gOwnerB.extractSubgraph(subGraphB);
536 if (!subGraphCloneA.isIsostructuralTo(subGraphCloneB))
537 return;
538 }
539
540 // Retain connection to parent to keep directionality of spanning tree!
541 // To this end, identify the seed of the subgraphs...
542 Vertex seedOnA = gOwnerA.getDeepestAmongThese(subGraphA);
543 Vertex seedOnB = gOwnerB.getDeepestAmongThese(subGraphB);
544 //...and ensure we use the same APs to link to the parent graph.
545 APMapping fixedRootAPs = new APMapping();
546 fixedRootAPs.put(seedOnA.getEdgeToParent().getTrgAP(),
547 seedOnB.getEdgeToParent().getTrgAP());
548
549 APMapFinder apmf = new APMapFinder(fragSpace,
550 allAPsA, needyAPsA, allAPsB, needyAPsB,
551 fixedRootAPs,
552 false, // false: we stop at the first good mapping
553 true, // true: only complete mapping
554 false); // false: free APs are not compatible by default
555 if (apmf.foundMapping())
556 {
557 XoverSite xos = new XoverSite(subGraphA, needyAPsA,
558 subGraphB, needyAPsB, xoverType);
559 if (!collector.contains(xos))
560 collector.add(xos);
561 }
562 }

577 private static boolean isCrossoverPossible(Edge eA, Edge eB,
578 FragmentSpace fragSpace)
579 {
580 APClass apClassSrcA = eA.getSrcAPClass();
581 APClass apClassTrgA = eA.getTrgAPClass();
582 APClass apClassSrcB = eB.getSrcAPClass();
583 APClass apClassTrgB = eB.getTrgAPClass();
584
585 if (apClassSrcA.isCPMapCompatibleWith(apClassTrgB, fragSpace))
586 {
587 if (apClassSrcB.isCPMapCompatibleWith(apClassTrgA, fragSpace))
588 {
589 return true;
590 }
591 }
592 return false;
593 }

737 + "vertex (AP "+chosenAPId+" of vertex "+vertex+").");
738 }
739 return extendLink(e, chosenNewVrtxId, mnt, fragSpace);

756 public static boolean extendLink(Edge edge, int chosenBBIdx,
757 Monitor mnt, FragmentSpace fragSpace) throws DENOPTIMException
758 {
759 //TODO: for reproducibility, the AP mapping should become an optional
760 // parameter: if given we try to use it, if not given we GraphLinkFinder

762
763 GraphLinkFinder glf = null;
764 if (chosenBBIdx < 0)
765 {
766 glf = new GraphLinkFinder(fragSpace, edge);

768 glf = new GraphLinkFinder(fragSpace, edge,chosenBBIdx);
769 }

775

777 // or the new link.

779 new LinkedHashMap<AttachmentPoint,Integer>();
780 for (Entry<AttachmentPoint, AttachmentPoint> e :

783 apMap.put(e.getKey(), e.getValue().getIndexInOwner());

785
786 DGraph graph = edge.getSrcAP().getOwner().getGraphOwner();
787 boolean done = graph.insertVertex(edge,
788 glf.getChosenAlternativeLink().getBuildingBlockId(),
789 glf.getChosenAlternativeLink().getBuildingBlockType(),
790 apMap, fragSpace);
791 if (!done)
792 {
793 mnt.increase(
794 CounterID.FAILEDMUTATTEMTS_PERFORM_NOADDLINK_EDIT);
795 }
796 return done;

822 protected static boolean rebuildBranch(Vertex vertex,
823 boolean force, int chosenVrtxIdx, int chosenApId,
824 GAParameters settings) throws DENOPTIMException
825 {
826 DGraph g = vertex.getGraphOwner();
827
828 // first get the edge with the parent
829 Edge e = vertex.getEdgeToParent();
830 if (e == null)
831 {
832 String msg = "Program Bug in substituteFragment: Unable to locate "
833 + "parent edge for vertex "+vertex+" in graph "+g;
834 settings.getLogger().log(Level.SEVERE, msg);
835 throw new DENOPTIMException(msg);
836 }
837
838 // vertex id of the parent
839 int pvid = e.getSrcVertex();
840 Vertex parentVrt = g.getVertexWithId(pvid);

843 boolean symmetry = g.hasSymmetryInvolvingVertex(vertex);
844
845 // delete the vertex and its children and all its symmetric partners
846 deleteFragment(vertex);
847
848 // extend the graph at this vertex but without recursion
849 return extendGraph(parentVrt,false,symmetry,force,chosenVrtxIdx,
850 chosenApId, settings);

994 protected static boolean extendGraph(Vertex curVrtx,
995 boolean extend,
996 boolean symmetryOnAps,
997 boolean force,
998 int chosenVrtxIdx,
999 int chosenApId,
1000 GAParameters settings)
1001 throws DENOPTIMException
1002 {
1003 // get settings //TODO: this should happen inside RunTimeParameters
1004 RingClosureParameters rcParams = new RingClosureParameters();
1005 if (settings.containsParameters(ParametersType.RC_PARAMS))
1006 {
1007 rcParams = (RingClosureParameters)settings.getParameters(
1008 ParametersType.RC_PARAMS);
1009 }
1010 FragmentSpaceParameters fsParams = new FragmentSpaceParameters();
1011 if (settings.containsParameters(ParametersType.FS_PARAMS))
1012 {
1013 fsParams = (FragmentSpaceParameters)settings.getParameters(
1014 ParametersType.FS_PARAMS);
1015 }
1016 int maxHeavyAtoms = fsParams.getMaxHeavyAtom();
1017
1018 // return true if the append has been successful
1019 boolean status = false;
1020
1021 // check if the fragment has available APs
1022 if (!curVrtx.hasFreeAP())
1023 {
1024 return status;
1025 }
1026
1027 int curVrtId = curVrtx.getVertexId();
1028 DGraph molGraph = curVrtx.getGraphOwner();
1029 int lvl = molGraph.getLevel(curVrtx);
1030
1031 ArrayList<Integer> addedVertices = new ArrayList<>();
1032
1033 ArrayList<AttachmentPoint> lstDaps = curVrtx.getAttachmentPoints();
1034 List<AttachmentPoint> toDoAPs = new ArrayList<AttachmentPoint>();
1035 toDoAPs.addAll(lstDaps);
1036 for (int i=0; i<lstDaps.size(); i++)
1037 {
1038 // WARNING: randomization decouples 'i' from the index of the AP
1039 // in the vertex's list of APs! So 'i' is just the i-th attempt on
1040 // the curVertex.
1041
1042 AttachmentPoint ap = settings.getRandomizer().randomlyChooseOne(
1043 toDoAPs);
1044 toDoAPs.remove(ap);
1045 int apId = ap.getIndexInOwner();
1046
1047 // is it possible to extend on this AP?
1048 if (!ap.isAvailable())
1049 {
1050 continue;
1051 }
1052
1053 // Ring closure does not change the size of the molecule, so we
1054 // give it an extra chance to occur irrespectively on molecular size
1055 // limit, while still subject of crowdedness probability.
1056 boolean allowOnlyRingClosure = false;
1057 if (!force)
1058 {
1059 // Decide whether we want to extend the graph at this AP?
1060 // Note that depending on the criterion (level/molSize) one
1061 // of these two first factors is 1.0.
1062 double molSizeProb = EAUtils.getMolSizeProbability(molGraph,
1063 settings);
1064 double byLevelProb = EAUtils.getGrowthByLevelProbability(lvl,
1065 settings);
1066 double crowdingProb = EAUtils.getCrowdingProbability(ap,
1067 settings);
1068 double extendGraphProb = molSizeProb * byLevelProb * crowdingProb;
1069 boolean fgrow = settings.getRandomizer().nextBoolean(
1070 extendGraphProb);
1071 if (!fgrow)
1072 {
1073 if (rcParams.allowRingClosures()
1074 && settings.getRandomizer().nextBoolean(byLevelProb
1075 * crowdingProb))
1076 {
1077 allowOnlyRingClosure = true;
1078 } else {
1079 continue;
1080 }
1081 }
1082 }
1083
1084 // Apply closability bias in selection of next fragment
1085 if (!allowOnlyRingClosure
1086 && rcParams.allowRingClosures()
1087 && rcParams.selectFragmentsFromClosableChains())
1088 {
1089 boolean successful = attachFragmentInClosableChain(curVrtx,
1090 apId, molGraph, addedVertices, settings);
1091 if (successful)
1092 {
1093 continue;
1094 }
1095 }
1096
1097 // find a compatible combination of fragment and AP
1098 IdFragmentAndAP chosenFrgAndAp = null;
1099 if (allowOnlyRingClosure)
1100 {
1101 chosenFrgAndAp = getRCVForSrcAp(curVrtx, apId,
1102 fsParams.getFragmentSpace());
1103 } else {
1104 chosenFrgAndAp = getFrgApForSrcAp(curVrtx,
1105 apId, chosenVrtxIdx, chosenApId, fsParams.getFragmentSpace());
1106 }
1107 int fid = chosenFrgAndAp.getVertexMolId();
1108 if (fid == -1)
1109 {
1110 continue;
1111 }
1112
1113 // Stop if graph is already too big
1114 Vertex incomingVertex = Vertex.newVertexFromLibrary(-1,
1115 chosenFrgAndAp.getVertexMolId(),
1116 BBType.FRAGMENT,
1117 fsParams.getFragmentSpace());
1118 if ((curVrtx.getGraphOwner().getHeavyAtomsCount() +
1119 incomingVertex.getHeavyAtomsCount()) > maxHeavyAtoms)
1120 {
1121 continue;
1122 }
1123
1124 // Decide on symmetric substitution within this vertex...
1125 boolean cpOnSymAPs = applySymmetry(
1126 fsParams.getFragmentSpace().imposeSymmetryOnAPsOfClass(
1127 ap.getAPClass()),
1128 settings.getSymmetryProbability(),
1129 fsParams.getRandomizer());
1130 SymmetricSet symAPs = curVrtx.getSymmetricAPs(apId);
1131 if (symAPs!=null
1132 && (cpOnSymAPs || symmetryOnAps)
1133 && !allowOnlyRingClosure)
1134 {
1135 // Are symmetric APs rooted on same atom?
1136 boolean allOnSameSrc = true;
1137 for (Integer symApId : symAPs.getList())
1138 {
1139 if (!curVrtx.getAttachmentPoints().get(symApId)
1140 .hasSameSrcAtom(ap))
1141 {
1142 allOnSameSrc = false;
1143 break;
1144 }
1145 }
1146
1147 if (allOnSameSrc)
1148 {
1149 // If the APs are rooted on the same src atom, we want to
1150 // apply the crowdedness probability to avoid over crowded
1151 // atoms
1152
1153 int crowdedness = EAUtils.getCrowdedness(ap);
1154
1155 SymmetricSet toKeep = new SymmetricSet();
1156
1157 // Start by keeping "ap"
1158 toKeep.add(apId);
1159 crowdedness = crowdedness + 1;
1160
1161 // Pick the accepted value once (used to decide how much
1162 // crowdedness we accept)
1163 double shot = settings.getRandomizer().nextDouble();
1164
1165 // Keep as many as allowed by the crowdedness decision
1166 for (Integer symApId : symAPs.getList())
1167 {
1168 if (symApId.compareTo(apId) == 0)
1169 continue;
1170
1171 double crowdProb = EAUtils.getCrowdingProbability(
1172 crowdedness, settings);
1173
1174 if (shot > crowdProb)
1175 break;
1176
1177 toKeep.add(symApId);
1178 crowdedness = crowdedness + 1;
1179 }
1180
1181 // Adjust the list of symmetric APs to work with
1182 symAPs = toKeep;
1183 }
1184 } else {
1185 symAPs = new SymmetricSet();
1186 symAPs.add(apId);
1187 }
1188
1189 // ...and inherit symmetry from previous levels
1190 boolean cpOnSymVrts = molGraph.hasSymmetryInvolvingVertex(curVrtx);
1191 SymmetricSet symVerts = new SymmetricSet();
1192 if (cpOnSymVrts)
1193 {
1194 symVerts = molGraph.getSymSetForVertexID(curVrtId);
1195 }
1196 else
1197 {
1198 symVerts.add(curVrtId);
1199 }
1200
1201 // Consider size after application of symmetry
1202 if ((curVrtx.getGraphOwner().getHeavyAtomsCount() +
1203 incomingVertex.getHeavyAtomsCount() * symVerts.size()
1204 * symAPs.size()) > maxHeavyAtoms)
1205 {
1206 continue;
1207 }
1208
1209 GraphUtils.ensureVertexIDConsistency(molGraph.getMaxVertexId());
1210
1211 // loop on all symmetric vertices, but can be only one.
1212 SymmetricSet newSymSetOfVertices = new SymmetricSet();
1213 for (Integer parVrtId : symVerts.getList())
1214 {
1215 Vertex parVrt = molGraph.getVertexWithId(parVrtId);
1216
1217 for (int si=0; si<symAPs.size(); si++)
1218 {
1219 int symApId = symAPs.get(si);
1220 AttachmentPoint symAP = parVrt.getAttachmentPoints()
1221 .get(symApId);
1222
1223 if (!symAP.isAvailable())
1224 {
1225 continue;
1226 }
1227
1228 // Finally add the fragment on a symmetric AP
1229 int newVrtId = GraphUtils.getUniqueVertexIndex();
1230 Vertex fragVertex = Vertex.newVertexFromLibrary(newVrtId,
1231 chosenFrgAndAp.getVertexMolId(),
1232 BBType.FRAGMENT,
1233 fsParams.getFragmentSpace());
1234 AttachmentPoint trgAP = fragVertex.getAP(
1235 chosenFrgAndAp.getApId());
1236
1237 molGraph.appendVertexOnAP(symAP, trgAP);
1238
1239 addedVertices.add(newVrtId);
1240 newSymSetOfVertices.add(newVrtId);
1241 }
1242 }
1243
1244 // If any, store symmetry of new vertices in the graph
1245 if (newSymSetOfVertices.size() > 1)
1246 {
1247 molGraph.addSymmetricSetOfVertices(newSymSetOfVertices);
1248 }
1249 } // end loop over APs
1250
1251 if (extend)
1252 {
1253 // attempt to further extend each of the newly added vertices
1254 for (int i=0; i<addedVertices.size(); i++)
1255 {
1256 int vid = addedVertices.get(i);
1257 Vertex v = molGraph.getVertexWithId(vid);
1258 extendGraph(v, extend, symmetryOnAps, settings);
1259 }
1260 }
1261
1262 if (addedVertices.size() > 0)
1263 status = true;
1264
1265 return status;
1266 }

1440 {
1441 addedVertices.add(newvid);
1442 // update list of candidate closable chains
1443 molGraph.getClosableChains().removeAll(
1444 chosenFfCc.getIncompatibleCC());
1445 APClass apc = curVertex.getAttachmentPoints().get(

1448 fsParams.getFragmentSpace().imposeSymmetryOnAPsOfClass(apc),

1452//TODO: implement symmetric substitution with closability bias

1459 settings.getLogger().log(Level.SEVERE, msg);

1528 protected static ArrayList<FragForClosabChains> getFragmentForClosableChain(
1529 Vertex curVertex,
1530 int dapidx,
1531 DGraph molGraph)
1532 throws DENOPTIMException
1533 {
1534 // Select candidate fragments respecting the closability conditions
1535 ArrayList<FragForClosabChains> lstChosenFfCc =
1536 new ArrayList<FragForClosabChains>();
1537
1538 if (molGraph.getClosableChains().size() == 0)
1539 {
1540 return lstChosenFfCc;
1541 }
1542
1543 if (curVertex.getBuildingBlockType() == BBType.SCAFFOLD)
1544 {
1545 for (ClosableChain cc : molGraph.getClosableChains())
1546 {
1547 int posInCc = cc.involvesVertex(curVertex);
1548 ChainLink cLink = cc.getLink(posInCc);
1549 int nfid = -1;
1550 BBType nfty = BBType.UNDEFINED;
1551 int nfap = -1;
1552
1553 if (cLink.getApIdToLeft() != dapidx &&
1554 cLink.getApIdToRight() != dapidx)
1555 {
1556 // Chain does not involve AP dapidx
1557 continue;
1558 }
1559
1560 if (cLink.getApIdToRight() == dapidx)
1561 {
1562 if (cc.getSize() > (posInCc+1))
1563 {
1564 // cLink is NOT the rightmost chain link
1565 ChainLink nextChainLink = cc.getLink(posInCc+1);
1566 nfid = nextChainLink.getIdx();
1567 nfty = nextChainLink.getFragType();
1568 nfap = nextChainLink.getApIdToLeft();
1569 }
1570 else
1571 {
1572 // cLink is the rightmost chain link
1573 // closability bias suggest NO fragment
1574 }
1575 }
1576 else if (cLink.getApIdToLeft() == dapidx)
1577 {
1578 if ((posInCc-1) >= 0)
1579 {
1580 // cLink is NOT the leftmost chain link
1581 ChainLink nextChainLink = cc.getLink(posInCc-1);
1582 nfid = nextChainLink.getIdx();
1583 nfty = nextChainLink.getFragType();
1584 nfap = nextChainLink.getApIdToRight();
1585 }
1586 else
1587 {
1588 // cLink is the leftmost chain link
1589 // closability bias suggest NO fragment
1590 }
1591 }
1592
1593 ArrayList<Integer> eligibleFrgId = new ArrayList<Integer>();
1594 eligibleFrgId.add(nfid);
1595 eligibleFrgId.add(nfty.toOldInt());
1596 eligibleFrgId.add(nfap);
1597 boolean found = false;
1598 for (FragForClosabChains ffcc : lstChosenFfCc)
1599 {
1600 int fidA = ffcc.getFragIDs().get(0);
1601 BBType ftyA = BBType.parseInt(ffcc.getFragIDs().get(1));
1602 int fapA = ffcc.getFragIDs().get(2);
1603 if (nfid==fidA && nfty==ftyA && nfap==fapA)
1604 {
1605 found = true;
1606 ffcc.getCompatibleCC().add(cc);
1607 }
1608 else
1609 {
1610 ffcc.getIncompatibleCC().add(cc);
1611 }
1612 }
1613 if (!found)
1614 {
1615 ArrayList<ClosableChain> compatChains =
1616 new ArrayList<ClosableChain>();
1617 ArrayList<ClosableChain> incompatChains =
1618 new ArrayList<ClosableChain>();
1619 for (FragForClosabChains otherFfCc : lstChosenFfCc)
1620 {
1621 incompatChains.addAll(otherFfCc.getCompatibleCC());
1622 }
1623 FragForClosabChains newChosenCc = new FragForClosabChains(
1624 compatChains,
1625 incompatChains,
1626 eligibleFrgId);
1627
1628 newChosenCc.addCompatibleCC(cc);
1629 lstChosenFfCc.add(newChosenCc);
1630 }
1631 }
1632 }
1633 else
1634 {
1635 Vertex parent = molGraph.getParent(curVertex);
1636 Edge edge = molGraph.getEdgeWithParent(
1637 curVertex.getVertexId());
1638 int prntId = parent.getBuildingBlockId();
1639 BBType prntTyp = parent.getBuildingBlockType();
1640 int prntAp = edge.getSrcAPID();
1641 int chidAp = edge.getTrgAPID();
1642 for (ClosableChain cc : molGraph.getClosableChains())
1643 {
1644 int posInCc = cc.involvesVertexAndAP(curVertex, dapidx, chidAp);
1645
1646 if (posInCc == -1)
1647 {
1648 // closable chain does not span this combination
1649 // of vertex and APs
1650 continue;
1651 }
1652
1653 ChainLink cLink = cc.getLink(posInCc);
1654 int nfid = -1;
1655 BBType nfty = BBType.UNDEFINED;
1656 int nfap = -1;
1657
1658 List<Integer> altertnativeDirections = new ArrayList<>();
1659 altertnativeDirections.add(-1);
1660 altertnativeDirections.add(+1);
1661 for (int altDir : altertnativeDirections)
1662 {
1663 ChainLink parentLink = cc.getLink(posInCc + altDir);
1664 int pLnkId = parentLink.getIdx();
1665 BBType pLnkTyp = parentLink.getFragType();
1666
1667 int pLnkAp = -1;
1668 int cApId = -1;
1669 if (altDir>0)
1670 {
1671 pLnkAp = parentLink.getApIdToRight();
1672 cApId = cLink.getApIdToLeft();

1755 public static boolean performCrossover(XoverSite site,
1756 FragmentSpace fragSpace) throws DENOPTIMException
1757 {
1758 DGraph gA = site.getA().get(0).getGraphOwner();
1759 DGraph gB = site.getB().get(0).getGraphOwner();
1760
1761 // All the APs that point away from the subgraph
1762 List<AttachmentPoint> allAPsOnA = gA.getSubgraphAPs(site.getA());
1763 List<AttachmentPoint> allAPsOnB = gB.getSubgraphAPs(site.getB());
1764
1765 // The APs that are required to have a mapping for proper crossover,
1766 // eg. because the change of subgraph needs to retain a super structure.
1767 List<AttachmentPoint> needyAPsOnA = site.getAPsNeedingMappingA();
1768 List<AttachmentPoint> needyAPsOnB = site.getAPsNeedingMappingB();
1769
1770 // APs that connects the subgraphs' root to the parent vertex
1771 AttachmentPoint apToParentA = null;
1772 AttachmentPoint apToParentB = null;
1773 for (AttachmentPoint ap : needyAPsOnA)
1774 {
1775 if (!ap.isSrcInUserThroughout())
1776 {
1777 apToParentA = ap;
1778 //WARNING: we assume there is one AND only one AP to parent!!!
1779 break;
1780 }
1781 }
1782 for (AttachmentPoint ap : needyAPsOnB)
1783 {
1784 if (!ap.isSrcInUserThroughout())
1785 {
1786 apToParentB = ap;
1787 //WARNING: we assume there is one AND only one AP to parent!!!
1788 break;
1789 }
1790 }
1791 if (apToParentA==null || apToParentB==null)
1792 {
1793 throw new DENOPTIMException("Could not identify any attachment "
1794 + "point connecting a subgraph to the rest of the graph. "
1795 + "This violates assumption that crossover does not "
1796 + "involve scaffold or vertexes without parent.");
1797 }
1798 // Check if the subgraphs can be used with reversed edge direction, or
1799 // bias the AP mapping to use the original source vertexes.
1800 APMapping fixedRootAPs = null;
1801 //TODO: REACTIVATE ONCE REVERSION of the subgrsph's spanning tree is in place.
1802 //if (!subG_M.isReversible() || !subG_F.isReversible())
1803 if (true)
1804 {
1805 // Constrain the AP mapping so that the AP originally used to
1806 // connect to the parent vertex, will also be used the same way.
1807 // Thus, forces retention of edge direction within the subgraph.
1808 fixedRootAPs = new APMapping();
1809 fixedRootAPs.put(apToParentA, apToParentB);
1810 }
1811
1812 // Find an AP mapping that satisfies both root-vertex constrain and
1813 // need to ensure the mapping of needy APs
1814 APMapFinder apmf = new APMapFinder(fragSpace,
1815 allAPsOnA, needyAPsOnA,
1816 allAPsOnB, needyAPsOnB, fixedRootAPs,
1817 false, // false means stop at the first compatible mapping.
1818 false, // false means we do not require complete mapping.
1819 false); // false means free AP are not considered compatible.
1820 if (!apmf.foundMapping())
1821 {
1822 // Since the xover site has been detected by searching for compatible
1823 // sites that do have a mapping there should always be at least one
1824 // mapping and this exception should never occur.
1825 throw new DENOPTIMException("Missing AP mapping for known XoverSite.");
1826 }
1827
1828 // To replace each subgraph in the original graphs, we need to
1829 // map the APs on the original A/B graph with those in the
1830 // corresponding incoming subgraph, which are clones of the original:
1831 DGraph subGraphA = gA.extractSubgraph(site.getA());
1832 DGraph subGraphB = gB.extractSubgraph(site.getB());
1833
1834 // Here we create the two AP mappings we need: one for A other for B.
1835 LinkedHashMap<AttachmentPoint,AttachmentPoint>
1836 apMapA = new LinkedHashMap<AttachmentPoint,AttachmentPoint>();
1837 LinkedHashMap<AttachmentPoint,AttachmentPoint>
1838 apMapB = new LinkedHashMap<AttachmentPoint,AttachmentPoint>();
1839 for (Map.Entry<AttachmentPoint, AttachmentPoint> e :
1840 apmf.getChosenAPMapping().entrySet())
1841 {
1842 AttachmentPoint apOnA = e.getKey();
1843 AttachmentPoint apOnB = e.getValue();
1844
1845 // NB: assumption that vertex IDs are healthy, AND that order of APs
1846 // is retained upon cloning of the subgraph!
1847 AttachmentPoint apOnSubGraphA = subGraphA.getVertexWithId(
1848 apOnA.getOwner().getVertexId()).getAP(
1849 apOnA.getIndexInOwner());
1850 AttachmentPoint apOnSubGraphB = subGraphB.getVertexWithId(
1851 apOnB.getOwner().getVertexId()).getAP(
1852 apOnB.getIndexInOwner());
1853 apMapA.put(apOnA, apOnSubGraphB);
1854 apMapB.put(apOnB, apOnSubGraphA);
1855 }
1856
1857 // Now we do the actual replacement of subgraphs
1858 if (!gA.replaceSubGraph(site.getA(), subGraphB, apMapA, fragSpace))
1859 return false;
1860 if (!gB.replaceSubGraph(site.getB(), subGraphA, apMapB, fragSpace))
1861 return false;
1862
1863 return true;
1864 }
