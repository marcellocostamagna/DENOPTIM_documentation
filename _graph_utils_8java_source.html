77 public static synchronized void resetUniqueVertexCounter(int val)
78 throws DENOPTIMException
79 {
80 if (vertexCounter.get() >= val)
81 {
82 String msg = "Attempt to reset the unique vertex ID using "
83 + val + " while the current value is "
84 + vertexCounter.get();
85 throw new DENOPTIMException(msg);
86 }
87 vertexCounter = new AtomicInteger(val);
88 }

98 public static synchronized int getUniqueVertexIndex()
99 {

101 throw new Error("Reached maximum value for "
102 + "Vertex identifier. Please contact the authors to "
103 + "request use of 'long' IDs.");

238 public static void writeSDFFields(IAtomContainer iac, DGraph g)
239 throws DENOPTIMException
240 {
241 iac.setProperty(DENOPTIMConstants.GCODETAG, g.getGraphId());
242 iac.setProperty(DENOPTIMConstants.GRAPHTAG, g.toString());
243 iac.setProperty(DENOPTIMConstants.GRAPHJSONTAG, g.toJson());
244 if (g.getLocalMsg() != null
245 && !g.getLocalMsg().toString().equals(""))
246 {
247 iac.setProperty(DENOPTIMConstants.GMSGTAG,g.getLocalMsg());
248 }
249 }

253 public static String getLabel(Vertex v)
254 {
255 if (!v.hasProperty("Label"))
256 return "";
257 return v.getGraphOwner().getGraphId()+"@"+v.getProperty(
258 "Label").toString();
