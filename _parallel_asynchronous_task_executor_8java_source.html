78 public ParallelAsynchronousTaskExecutor(int numberOfTasks, Logger logger)
79 {
80 this.logger = logger;
81 futures = new ArrayList<>();
82 submitted = new ArrayList<>();
83
84 tpe = new ThreadPoolExecutor(numberOfTasks,
85 numberOfTasks,
86 Long.MAX_VALUE,
87 TimeUnit.NANOSECONDS,
88 new ArrayBlockingQueue<Runnable>(1));
89
90 Runtime.getRuntime().addShutdownHook(new Thread()
91 {
92 @Override
93 public void run()
94 {
95 tpe.shutdown(); // Disable new tasks from being submitted
96 try
97 {
98 // Wait a while for existing tasks to terminate
99 if (!tpe.awaitTermination(30, TimeUnit.SECONDS))
100 {
101 tpe.shutdownNow(); // Cancel currently executing tasks
102 }
103
104 if (!tpe.awaitTermination(60, TimeUnit.SECONDS))
105 {
106 // pool didn't terminate after the second try
107 }
108 }
109 catch (InterruptedException ie)
110 {
111 cleanup(tpe, futures, submitted);
112 // (Re-)Cancel if current thread also interrupted
113 tpe.shutdownNow();
114 // Preserve interrupt status
115 Thread.currentThread().interrupt();
116 }
117 }
118 });
119
120 // by default the ThreadPoolExecutor will throw an exception
121 tpe.setRejectedExecutionHandler(new RejectedExecutionHandler()
122 {
123 @Override
124 public void rejectedExecution(Runnable r,
125 ThreadPoolExecutor executor)
126 {
127 try
128 {
129 // this will block if the queue is full
130 executor.getQueue().put(r);
131 }
132 catch (InterruptedException ex)
133 {
134 ex.printStackTrace();
135 String msg = "EXCEPTION in rejectedExecution.";
136 logger.log(Level.WARNING,msg);
137 }
138 }
139 });
140 }

212 public void run() throws DENOPTIMException, IOException
213 {
214 String msg = "";
215 StopWatch watch = new StopWatch();
216 watch.start();
217
218 if (!doPreFlightOperations())
219 return;
220
221 // Start parallel threads for generating and/or manipulating fragments
222 tpe.prestartAllCoreThreads();
223
224 createAndSubmitTasks();
225
226 // This sounds weird when reading the doc, but the following does wait
227 // for the threads to complete.
228 // TODO try ExecutorCompletionService
229
230 // shutdown thread pool
231 tpe.shutdown();
232 try
233 {
234 // wait a bit for pending tasks to finish
235 while (!tpe.awaitTermination(5, TimeUnit.SECONDS))
236 {
237 // do nothing
238 }
239 }
240 catch (InterruptedException ex)
241 {
242 //Do nothing
243 }
244
245 if (!doPostFlightOperations())
246 return;
247
248 // closing messages
249 watch.stop();
250 msg = "Overall time: " + watch.toString() + ". "
251 + DENOPTIMConstants.EOL
252 + this.getClass().getSimpleName() + " run completed."
253 + DENOPTIMConstants.EOL;
254 logger.log(Level.INFO, msg);
255 }

259 protected void submitTask(Task task, String logFilePathname)
260 {
261 submitted.add(task);
262 futures.add(tpe.submit(task));
263 String msg = task.getClass().getSimpleName() + " "
264 + task.getId() + " submitted.";
265 if (logFilePathname!=null && !logFilePathname.isBlank())
266 msg = msg + " Log file: " + logFilePathname;
267 logger.log(Level.INFO, msg);

292 private void cleanup(ThreadPoolExecutor tpe, List<Future<Object>> futures,
293 ArrayList<Task> submitted)
294 {
295 for (Future<Object> f : futures)
296 {
297 f.cancel(true);
298 }
299
300 for (Task tsk: submitted)
301 {
302 tsk.stopTask();
303 }
304 submitted.clear();
305 tpe.getQueue().clear();
306 }
