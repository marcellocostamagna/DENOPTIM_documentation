322 protected String getStringFromTable(String key, JTable tab, int[] fields,
323 String sep, boolean skipHeader)
324 {
325 StringBuilder sb = new StringBuilder();
326 int firstRow = 0;
327 if (skipHeader)
328 {
329 firstRow = 1;
330 }
331 for (int i=firstRow; i<tab.getRowCount(); i++)
332 {
333 String val = "";
334 for (int j=0; j<fields.length; j++)
335 {
336 if (j==0)
337 {
338 val = tab.getValueAt(i, fields[j]).toString();
339 }
340 else
341 {
342 val = val + sep + tab.getValueAt(i, fields[j]).toString();
343 }
344 }
345 sb.append(key).append("=").append(val).append(NL);
346 }
347 return sb.toString();
348 }

371 protected void importParametersFromDenoptimParamsFile(String fileName,
372 String keyRoot) throws Exception
373 {
374 if (keyRoot.equals("") || keyRoot == null)
375 {
376 throw new Exception("<html>Root of DENOPTIM keywords not defined."
377 + "<br>Bugus use of method "
378 + "'importParametersFromDenoptimParamsFile'!</html>");
379 }
380 String parType = keyRoot.replaceAll("-", "");
381 if (fileName.equals("") || !FileUtils.checkExists(fileName))
382 {
383 throw new Exception("<html>Source file for the " + parType
384 + " parameters is not found!<br>Please, specify another"
385 + " file name.");
386 }
387
388 preliminatyTasksUponImportingParams();
389
390 String line;
391 BufferedReader br = null;
392 try
393 {
394 br = new BufferedReader(new FileReader(fileName));
395 while ((line = br.readLine()) != null)
396 {
397 if ((line.trim()).length() == 0)
398 {
399 continue;
400 }
401
402 if (line.startsWith("#"))
403 {
404 continue;
405 }
406
407 for (String flag : checkedFlags.keySet())
408 {
409 if (line.toUpperCase().startsWith(flag.toUpperCase()))
410 checkedFlags.put(flag,true);
411 }
412
413 if (line.toUpperCase().startsWith(keyRoot.toUpperCase()))
414 {
415 importParameterFromLine(line);
416 }
417 }
418 }
419 catch (IOException ioe)
420 {
421 throw new Exception("Unable to read file '" + fileName + "'", ioe);
422 }
423 finally
424 {
425 try
426 {
427 if (br != null)
428 {
429 br.close();
430 br = null;
431 }
432 }
433 catch (IOException ioe)
434 {
435 throw new Exception("Unable to close file '" + fileName + "'",
436 ioe);
437 }
438 }
439
440 //adapt visibility to imported parameters
441 adaptVisibility();
442 }

466 protected void importParameterFromLine(String line) throws Exception
467 {
468 String key = line.trim();
469 String value = "";
470 if (line.contains("="))
471 {
472 key = line.substring(0,line.indexOf("=")).trim();
473 value = line.substring(line.indexOf("=") + 1).trim();
474 }
475 importSingleParameter(key,value);
