203 private LinkedHashMap<Candidate,

209 public XoverSitesAmongCandidates()
210 {
211 data = new LinkedHashMap<Candidate,LinkedHashMap<Candidate,
212 List<XoverSite>>>();

214
223 public void put(Candidate c1, Candidate c2,
224 List<XoverSite> xoversite)
225 {
226 if (data.containsKey(c1))
227 {
228 data.get(c1).put(c2, xoversite);
229 } else {
230 LinkedHashMap<Candidate,List<XoverSite>> toC1 =
231 new LinkedHashMap<Candidate,List<XoverSite>>();
232 toC1.put(c2, xoversite);
233 data.put(c1, toC1);
234 }
235
236 List<XoverSite> revPairs =
237 new ArrayList<XoverSite>();
238 for (XoverSite pair : xoversite)
239 {
240 XoverSite revPair = pair.createMirror();
241 revPairs.add(revPair);
242 }
243
244 if (data.containsKey(c2))
245 {
246 data.get(c2).put(c1, revPairs);
247 } else {
248 LinkedHashMap<Candidate,List<XoverSite>> toC2 =
249 new LinkedHashMap<Candidate,List<XoverSite>>();
250 toC2.put(c1, revPairs);
251 data.put(c2, toC2);
252 }
253 }

328 public XoverSitesAmongCandidates clone()
329 {
330 XoverSitesAmongCandidates cloned = new XoverSitesAmongCandidates();
331 for (Candidate c1 : data.keySet())
332 {
333 LinkedHashMap<Candidate, List<XoverSite>> inner =
334 new LinkedHashMap<Candidate, List<XoverSite>>();
335 for (Candidate c2 : data.get(c1).keySet())
336 {
337 List<XoverSite> lst = new ArrayList<XoverSite>();
338 for (XoverSite arr : data.get(c1).get(c2))
339 {
340 lst.add(arr.clone());
341 }
342 inner.put(c2,lst);
343 }
344 cloned.data.put(c1, inner);
345 }
346 return cloned;
347 }

361 public ArrayList<Candidate> getXoverPartners(Candidate memberA,
362 ArrayList<Candidate> eligibleParents, FragmentSpace fragSpace)
363 {
364 DGraph gA = memberA.getGraph();
365
366 // Update to make sure we cover any combination of members that has not
367 // been considered before
368 for (Candidate memberB : eligibleParents)
369 {
370 if (memberA == memberB)
371 {
372 continue;
373 }
374
375 if (xoverCompatibilities.contains(memberA, memberB))
376 {
377 continue;
378 }
379
380 DGraph gB = memberB.getGraph();
381
382 if (gA.sameAs(gB, new StringBuilder()))
383 continue;
384
385 try
386 {
387 List<XoverSite> xoverSites = GraphOperations
388 .locateCompatibleXOverPoints(gA, gB, fragSpace);
389 xoverCompatibilities.put(memberA, memberB, xoverSites);
390 } catch (DENOPTIMException e)
391 {
392 settings.getLogger().log(Level.FINE, "Could not identify "
393 + "crossover sites between " + memberA.getName()
394 + " and " + memberB.getName() + ".");
395 e.printStackTrace();
396 }
397 }
398 return xoverCompatibilities.getMembersCompatibleWith(memberA);
399 }

474 public boolean isWithinPercentile(double value, double percentile)
475 {
476 double min = getMinFitness();
477 double max = getMaxFitness();
478 double threshold = (1.0 - percentile) * (max-min);
479
480 if (value > (threshold+min))
481 return true;
482 else
