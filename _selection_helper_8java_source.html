51 protected static Candidate[] performTournamentSelection(
52 ArrayList<Candidate> population, int sz, RunTimeParameters settings)
53 {
54 Candidate[] selection = new Candidate[sz];
55 for (int i=0; i<sz; i++)
56 {
57 // Pick two candidates at random.
58 Candidate p1 = settings.getRandomizer().randomlyChooseOne(population);
59 Candidate p2 = settings.getRandomizer().randomlyChooseOne(population);
60
61 // Use a random value to decide weather to select the fitter individual
62 // or the weaker one.
63 boolean selectFitter = settings.getRandomizer().nextBoolean();
64
65 if (selectFitter)
66 {
67 // Select the fitter candidate.
68 selection[i] = p1.getFitness() > p2.getFitness() ? p1 : p2;
69 }
70 else
71 {
72 // Select the weaker candidate.
73 selection[i] = p2.getFitness() > p1.getFitness() ? p1 : p2;
74 }
75 }
76 return selection;
77 }

109 protected static Candidate[] performSUS(ArrayList<Candidate> population,
110 int sz, RunTimeParameters settings)
111 {
112 int k = population.size();
113 Candidate[] selection = new Candidate[sz];
114 // Calculate the sum of all fitness values.
115 double aggregateFitness = 0;
116
117 for (int i=0; i<k; i++)
118 {
119 aggregateFitness += population.get(i).getFitness();
120 }
121
122
123 // Pick a random offset between 0 and 1 as the starting point for selection.
124 double startOffset = settings.getRandomizer().nextDouble();
125 double cumulativeExpectation = 0;
126 int index = 0;
127 int c = 0;
128 for (int i=0; i<k; i++)
129 {
130 // Calculate the number of times this candidate is expected to
131 // be selected on average and add it to the cumulative total
132 // of expected frequencies.
133 cumulativeExpectation += population.get(i).getFitness()
134 / aggregateFitness * sz;
135
136 // If f is the expected frequency, the candidate will be selected at
137 // least as often as floor(f) and at most as often as ceil(f). The
138 // actual count depends on the random starting offset.
139 while (cumulativeExpectation > startOffset + index)
140 {
141 selection[c] = population.get(i);
142 c++;
143 index++;
144 }
145 }
146
147 return selection;
148 }

166 protected static Candidate[] performRWS(ArrayList<Candidate> population,
167 int sz, RunTimeParameters settings)
168 {
169 int k = population.size();
170 Candidate[] selection = new Candidate[sz];
171
172 double[] cumulativeFitnesses = new double[k];
173 cumulativeFitnesses[0] = population.get(0).getFitness();
174
175 for (int i=1; i<k; i++)
176 {
177 double fitness = population.get(i).getFitness();
178
179 cumulativeFitnesses[i] = cumulativeFitnesses[i-1] + fitness;
180 }
181
182 for (int i=0; i<sz; i++)
183 {
184 double randomFitness = settings.getRandomizer().nextDouble() *
185 cumulativeFitnesses[cumulativeFitnesses.length-1];
186 int index = Arrays.binarySearch(cumulativeFitnesses, randomFitness);
187 if (index < 0)
188 {
189 // Convert negative insertion point to array index.
190 index = Math.abs(index + 1);
191 }
192 selection[i] = population.get(index);
193 }
194
195 return selection;
196 }
