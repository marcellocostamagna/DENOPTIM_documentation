316 public void loadVertexToViewer(Vertex v)
317 {
318 vertex = v;
319 if (v instanceof Fragment) {
320 labTitle.setText("Fragment");
321 Fragment frag = (Fragment) v;
322 loadFragmentToViewer(frag);
323 } else if (v instanceof EmptyVertex) {
324 labTitle.setText("EmptyVertex");
325 EmptyVertex ev = (EmptyVertex) v;
326 loadEmptyVertexToViewer(ev);
327 } else if (v instanceof Template) {
328 labTitle.setText("Template");
329 Template tmpl = (Template) v;
330 loadTemplateToViewer(tmpl);
331 } else {
332 System.err.println("Loading empty card as a result of vertex with "
333 + "type " + v.getClass().getName());
334 switchToEmptyCard();
335 }
336 }

381 private void loadTemplateToViewer(Template tmpl)
382 {
383 if (tmpl.containsAtoms())
384 {
385 Fragment frag;
386 try
387 {
388 frag = new Fragment(tmpl.getVertexId(),
389 tmpl.getIAtomContainer(), tmpl.getBuildingBlockType());
390
391 loadFragmentToViewer(frag);
392 btnSwitchToMolViewer.setEnabled(true);
393 btnSwitchToNodeViewer.setEnabled(true);
394 switchbleByVertexType = true;
395 } catch (DENOPTIMException e)
396 {
397 // We lease data in the viewer to increase speed of execution, but the
398 // leftover is outdated! This is the meaning of 'true'
399 fragViewer.clearAll(true);
400 switchbleByVertexType = false;
401 }
402 } else {
403 // We lease data in the viewer to increase speed of execution, but the
404 // leftover is outdated! This is the meaning of 'true'
405 fragViewer.clearAll(true);
406 switchbleByVertexType = false;
407 }
408 graphNodeViewer.loadVertexToViewer(tmpl);
409 switchToGraphNodeViewer();
410 graphNodeViewer.setVertexSpecificEditableAPTable(false);
411 }
